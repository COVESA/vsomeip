19void profile_04_checker::check(const e2e_buffer &_buffer, instance_t _instance,
20 e2e::profile_interface::check_status_t &_generic_check_status) {
21
22 std::lock_guard<std::mutex> lock(check_mutex_);
23 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_ERROR;
24
25 if (_instance > VSOMEIP_E2E_PROFILE04_MAX_INSTANCE) {
26 VSOMEIP_ERROR << "E2E Profile 4 can only be used for instances [1-255]";
27 return;
28 }
29
31 if (verify_input(_buffer)) {
33 uint16_t its_received_length;
34 if (read_16(_buffer, its_received_length, 0)) {
36 uint16_t its_received_counter;
37 if (read_16(_buffer, its_received_counter, 2)) {
39 uint32_t its_received_data_id;
40 if (read_32(_buffer, its_received_data_id, 4)) {
42 uint32_t its_received_crc;
43 if (read_32(_buffer, its_received_crc, 8)) {
44 uint32_t its_crc = profile_04::compute_crc(config_, _buffer);
46 if (its_received_crc != its_crc) {
47 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_WRONG_CRC;
48 VSOMEIP_ERROR << std::hex << "E2E P04 protection: CRC32 does not match: calculated CRC: "
49 << its_crc << " received CRC: " << its_received_crc;
50 } else {
51 uint32_t its_data_id(uint32_t(_instance) << 24 | config_.data_id_);
52 if (its_received_data_id == its_data_id
53 && static_cast<size_t>(its_received_length) == _buffer.size()
54 && verify_counter(its_received_counter)) {
55 _generic_check_status = e2e::profile_interface::generic_check_status::E2E_OK;
56 }
57 counter_ = its_received_counter;
58 }
59 }
60 }
61 }
62 }

74bool
75profile_04_checker::verify_counter(uint16_t _received_counter) const {
76
77 static bool has_counter(false);
78 uint16_t its_delta(0);
79
80 if (has_counter) {
81 if (counter_ < _received_counter)
82 its_delta = uint16_t(_received_counter - counter_);
83 else
84 its_delta = uint16_t(uint16_t(0xffff) - counter_ + _received_counter);
85 } else {
86 has_counter = true;
87 }
88
89 return (its_delta <= config_.max_delta_counter_);
90}
