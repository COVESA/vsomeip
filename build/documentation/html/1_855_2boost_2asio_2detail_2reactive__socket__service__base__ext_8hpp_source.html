177 // Send the given data to the peer.

182 {

184 ConstBufferSequence> bufs(buffers);

187 bufs.buffers(), bufs.count(), flags, bufs.all_empty(), ec);
188 }

193 {
194 // Wait for socket to become ready.
195 socket_ops::poll_write(impl.socket_, impl.state_, ec);
196
197 return 0;
198 }

200 // Start an asynchronous send. The data being sent must be valid for the
201 // lifetime of the asynchronous operation.
202 template <typename ConstBufferSequence, typename Handler>
203 void async_send(base_implementation_type& impl,
204 const ConstBufferSequence& buffers,
205 socket_base::message_flags flags, Handler& handler)
206 {
207 bool is_continuation =
208 boost_asio_handler_cont_helpers::is_continuation(handler);
209
210 // Allocate and construct an operation to wrap the handler.
211 typedef reactive_socket_send_op<ConstBufferSequence, Handler> op;
212 typename op::ptr p = { boost::asio::detail::addressof(handler),
213 boost_asio_handler_alloc_helpers::allocate(
214 sizeof(op), handler), 0 };
215 p.p = new (p.v) op(impl.socket_, buffers, flags, handler);
216
217 BOOST_ASIO_HANDLER_CREATION((p.p, "socket", &impl, "async_send"));
218
219 start_op(impl, reactor::write_op, p.p, is_continuation, true,
220 ((impl.state_ & socket_ops::stream_oriented)
221 && buffer_sequence_adapter<boost::asio::const_buffer,
222 ConstBufferSequence>::all_empty(buffers)));
223 p.v = p.p = 0;
224 }

245 p.v = p.p = 0;
246 }

271 // Start an asynchronous receive. The buffer for the data being received
272 // must be valid for the lifetime of the asynchronous operation.
273 template <typename MutableBufferSequence, typename Handler>
274 void async_receive(base_implementation_type& impl,
275 const MutableBufferSequence& buffers,
276 socket_base::message_flags flags, Handler& handler)
277 {
278 bool is_continuation =
279 boost_asio_handler_cont_helpers::is_continuation(handler);
280
281 // Allocate and construct an operation to wrap the handler.
282 typedef reactive_socket_recv_op_ext<MutableBufferSequence, Handler> op;

284 boost_asio_handler_alloc_helpers::allocate(
285 sizeof(op), handler), 0 };
286 p.p = new (p.v) op(impl.socket_, impl.state_, buffers, flags, handler);
287
288 BOOST_ASIO_HANDLER_CREATION((p.p, "socket", &impl, "async_receive"));
289
290 start_op(impl,
291 (flags & socket_base::message_out_of_band)
292 ? reactor::except_op : reactor::read_op,
293 p.p, is_continuation,

295 ((impl.state_ & socket_ops::stream_oriented)
296 && buffer_sequence_adapter<boost::asio::mutable_buffer,
297 MutableBufferSequence>::all_empty(buffers)));
298 p.v = p.p = 0;
299 }

340
341 // Wait until data can be received without blocking.
342 size_t receive_with_flags(base_implementation_type& impl,
343 const null_buffers&, socket_base::message_flags,
344 socket_base::message_flags& out_flags, boost::system::error_code& ec)
345 {
346 // Wait for socket to become ready.
347 socket_ops::poll_read(impl.socket_, impl.state_, ec);
348
349 // Clear out_flags, since we cannot give it any other sensible value when
350 // performing a null_buffers operation.
351 out_flags = 0;
352
353 return 0;
354 }
