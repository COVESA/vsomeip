96void serializer::reset() {
97 if (buffer_shrink_threshold_) {
98 if (data_.size() < (data_.capacity() >> 1)) {
99 shrink_count_++;
100 } else {
101 shrink_count_ = 0;
102 }
103 }
104 data_.clear();
105 if (buffer_shrink_threshold_ && shrink_count_ > buffer_shrink_threshold_) {
106 data_.shrink_to_fit();
107 shrink_count_ = 0;
108 }
109}
