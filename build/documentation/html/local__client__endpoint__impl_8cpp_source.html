76void local_client_endpoint_impl::stop() {
77 {
78 std::lock_guard<std::mutex> its_lock(mutex_);
79 sending_blocked_ = true;
80 }
81 {
82 std::lock_guard<std::mutex> its_lock(connect_timer_mutex_);
83 boost::system::error_code ec;
84 connect_timer_.cancel(ec);
85 }
86 connect_timeout_ = VSOMEIP_DEFAULT_CONNECT_TIMEOUT;
87
88 bool is_open(false);
89 {
90 std::lock_guard<std::mutex> its_lock(socket_mutex_);
91 is_open = socket_->is_open();
92 }
93 if (is_open) {
94 bool send_queue_empty(false);
95 std::uint32_t times_slept(0);
96
97 while (times_slept <= 50) {
98 mutex_.lock();
99 send_queue_empty = (queue_.size() == 0);
100 mutex_.unlock();
101 if (send_queue_empty) {
102 break;
103 } else {
104 std::this_thread::sleep_for(std::chrono::milliseconds(10));
105 times_slept++;
106 }
107 }
108 }
109 shutdown_and_close_socket(false);
110}

112void local_client_endpoint_impl::connect() {
113 boost::system::error_code its_connect_error;
114 {
115 std::lock_guard<std::mutex> its_lock(socket_mutex_);
116 boost::system::error_code its_error;
117 socket_->open(remote_.protocol(), its_error);
118
119 if (!its_error || its_error == boost::asio::error::already_open) {
120 socket_->set_option(boost::asio::socket_base::reuse_address(true), its_error);
121 if (its_error) {
122 VSOMEIP_WARNING << "local_client_endpoint_impl::connect: "
123 << "couldn't enable SO_REUSEADDR: " << its_error.message();
124 }
125 state_ = cei_state_e::CONNECTING;
126 socket_->connect(remote_, its_connect_error);
127
128// Credentials
129#ifndef _WIN32
130 if (!its_connect_error) {
131 auto its_host = endpoint_host_.lock();
132 if (its_host) {
133 credentials::send_credentials(socket_->native_handle(),
134 its_host->get_client());
135 }
136 } else {
137 VSOMEIP_WARNING << "local_client_endpoint::connect: Couldn't "
138 << "connect to: " << remote_.path() << " ("
139 << its_connect_error.message() << " / " << std::dec
140 << its_connect_error.value() << ")";
141 }
142#endif
143
144 } else {
145 VSOMEIP_WARNING << "local_client_endpoint::connect: Error opening socket: "
146 << its_error.message() << " (" << std::dec << its_error.value()
147 << ")";
148 its_connect_error = its_error;
149 }
150 }
151 // call connect_cbk asynchronously
152 try {
153 strand_.post(
154 std::bind(&client_endpoint_impl::connect_cbk, shared_from_this(),
155 its_connect_error));
156 } catch (const std::exception &e) {
157 VSOMEIP_ERROR << "local_client_endpoint_impl::connect: " << e.what();
158 }
159}

206void local_client_endpoint_impl::send_queued() {
207 static const byte_t its_start_tag[] = { 0x67, 0x37, 0x6D, 0x07 };
208 static const byte_t its_end_tag[] = { 0x07, 0x6D, 0x37, 0x67 };
209 std::vector<boost::asio::const_buffer> bufs;
210
211 message_buffer_ptr_t its_buffer;
212 if(queue_.size()) {
213 its_buffer = queue_.front();
214 } else {
215 return;
216 }
217
218 bufs.push_back(boost::asio::buffer(its_start_tag));
219 bufs.push_back(boost::asio::buffer(*its_buffer));
220 bufs.push_back(boost::asio::buffer(its_end_tag));
221
222 {
223 std::lock_guard<std::mutex> its_lock(socket_mutex_);
224 boost::asio::async_write(
225 *socket_,
226 bufs,
227 std::bind(
228 &client_endpoint_impl::send_cbk,
229 std::dynamic_pointer_cast<
230 local_client_endpoint_impl
231 >(shared_from_this()),
232 std::placeholders::_1,
233 std::placeholders::_2,
234 its_buffer

254void local_client_endpoint_impl::receive_cbk(
255 boost::system::error_code const &_error, std::size_t _bytes) {
256 if (_error) {
257 if (_error == boost::asio::error::operation_aborted) {
258 // endpoint was stopped
259 return;
260 } else if (_error == boost::asio::error::connection_reset
261 || _error == boost::asio::error::eof
262 || _error == boost::asio::error::bad_descriptor) {
263 VSOMEIP_TRACE << "local_client_endpoint:"
264 " connection_reseted/EOF/bad_descriptor";
265 } else if (_error) {
266 VSOMEIP_ERROR << "Local endpoint received message ("
267 << _error.message() << ")";
268 }
269 error_handler_t handler;
270 {
271 std::lock_guard<std::mutex> its_lock(error_handler_mutex_);
272 handler = error_handler_;
273 }
274 if (handler)
275 handler();
276 } else {
277
278#if 0
279 std::stringstream msg;
280 msg << "lce<" << this << ">::recv: ";
281 for (std::size_t i = 0; i < recv_buffer_.size(); i++)
282 msg << std::setw(2) << std::setfill('0') << std::hex
283 << (int)recv_buffer_[i] << " ";
284 VSOMEIP_INFO << msg.str();
285#endif
286
287 // We only handle a single message here. Check whether the message
288 // format matches what we do expect.
289 if (_bytes == VSOMEIP_ASSIGN_CLIENT_ACK_COMMAND_SIZE + 8
290 && recv_buffer_[0] == 0x67 && recv_buffer_[1] == 0x37
291 && recv_buffer_[2] == 0x6d && recv_buffer_[3] == 0x07
292 && recv_buffer_[4] == VSOMEIP_ASSIGN_CLIENT_ACK
293 && recv_buffer_[13] == 0x07 && recv_buffer_[14] == 0x6d
294 && recv_buffer_[15] == 0x37 && recv_buffer_[16] == 0x67) {
295
296 auto its_routing_host = routing_host_.lock();
297 if (its_routing_host)
298 its_routing_host->on_message(&recv_buffer_[4],
299 static_cast<length_t>(recv_buffer_.size() - 8), this);
300 }
301
302 receive();
303 }

349bool local_client_endpoint_impl::check_packetizer_space(std::uint32_t _size) {
350 if (train_.buffer_->size() + _size < train_.buffer_->size()) {
351 VSOMEIP_ERROR << "Overflow in packetizer addition ~> abort sending!";
352 return false;
353 }
354 if (train_.buffer_->size() + _size > max_message_size_
355 && !train_.buffer_->empty()) {
356 queue_.push_back(train_.buffer_);
357 queue_size_ += train_.buffer_->size();
358 train_.buffer_ = std::make_shared<message_buffer_t>();
359 }
360 return true;
361}

371bool local_client_endpoint_impl::send(const std::vector<byte_t>& _cmd_header,
372 const byte_t *_data, uint32_t _size) {
373 std::lock_guard<std::mutex> its_lock(mutex_);
374 bool ret(true);
375 const bool queue_size_zero_on_entry(queue_.empty());
376
377 const std::uint32_t its_complete_size = static_cast<std::uint32_t>(
378 _cmd_header.size() + _size);
379 if (endpoint_impl::sending_blocked_ ||
380 check_message_size(nullptr, its_complete_size) != cms_ret_e::MSG_OK ||
381 !check_packetizer_space(its_complete_size)||
382 !check_queue_limit(_data, its_complete_size)) {
383 ret = false;
384 } else {
385#if 0
386 std::stringstream msg;
387 msg << "lce::send: ";
388 for (uint32_t i = 0; i < _size; i++)
389 msg << std::hex << std::setw(2) << std::setfill('0')
390 << (int)_data[i] << " ";
391 VSOMEIP_INFO << msg.str();
392#endif
393 train_.buffer_->reserve(its_complete_size);
394 train_.buffer_->insert(train_.buffer_->end(), _cmd_header.begin(), _cmd_header.end());
395 train_.buffer_->insert(train_.buffer_->end(), _data, _data + _size);
396 queue_train(queue_size_zero_on_entry);
397 }
398 return ret;
399}
