75std::shared_ptr<local_server_endpoint_impl> endpoint_manager_base::create_local_server(
76 const std::shared_ptr<routing_host>& _routing_host) {
77 std::shared_ptr<local_server_endpoint_impl> its_server_endpoint;
78 std::stringstream its_path;
79 its_path << utility::get_base_path(configuration_) << std::hex << rm_->get_client();
80 const client_t its_client = rm_->get_client();
81#ifdef _WIN32
82 ::_unlink(its_path.str().c_str());
83 int port = VSOMEIP_INTERNAL_BASE_PORT + its_client;
84#else
85 if (-1 == ::unlink(its_path.str().c_str()) && errno != ENOENT) {
86 VSOMEIP_ERROR << "endpoint_manager_base::init_receiver unlink failed ("
87 << its_path.str() << "): "<< std::strerror(errno);
88 }
89#endif
90 try {
91 its_server_endpoint = std::make_shared<local_server_endpoint_impl>(
92 shared_from_this(), _routing_host,
93#ifdef _WIN32
94 boost::asio::ip::tcp::endpoint(boost::asio::ip::tcp::v4(), port),
95#else
96 boost::asio::local::stream_protocol_ext::endpoint(its_path.str()),
97#endif
98 io_,
99 configuration_, false);
100#ifdef _WIN32
101 VSOMEIP_INFO << "Listening at " << port;
102#else
103 VSOMEIP_INFO << "Listening at " << its_path.str();
104#endif
105 } catch (const std::exception &e) {
106 VSOMEIP_ERROR << "Local server endpoint creation failed. Client ID: "
107 << std::hex << std::setw(4) << std::setfill('0') << its_client
108#ifdef _WIN32
109 << " Port: " << std::dec << port
110#else
111 << " Path: " << its_path.str()
112#endif
113 << " Reason: " << e.what();
114 }
115 return its_server_endpoint;
116}

154void
155endpoint_manager_base::log_client_states() const {
156 std::vector<std::pair<client_t, size_t> > its_client_queue_sizes;
157 std::stringstream its_log;
158
159 {
160 std::lock_guard<std::mutex> its_lock(local_endpoint_mutex_);
161 for (const auto e : local_endpoints_) {
162 size_t its_queue_size = e.second->get_queue_size();
163 if (its_queue_size > VSOMEIP_DEFAULT_QUEUE_WARN_SIZE) {
164 its_client_queue_sizes.push_back(
165 std::make_pair(e.first, its_queue_size));
166 }
167 }
168 }
169
170 std::sort(its_client_queue_sizes.begin(), its_client_queue_sizes.end(),
171 [](const std::pair<client_t, size_t> &_a,
172 const std::pair<client_t, size_t> &_b) {
173 return (_a.second > _b.second);
174 });
175
176 size_t its_max(std::min(size_t(10), its_client_queue_sizes.size()));
177 for (size_t i = 0; i < its_max; i++) {
178 its_log << std::hex << std::setw(4) << std::setfill('0')
179 << its_client_queue_sizes[i].first << ":"
180 << std::dec << its_client_queue_sizes[i].second;
181 if (i < its_max-1)
182 its_log << ", ";
183 }
184
185 if (its_log.str().length() > 0)
186 VSOMEIP_WARNING << "ICQ: [" << its_log.str() << "]";
187}

189std::shared_ptr<endpoint> endpoint_manager_base::create_local_unlocked(client_t _client) {
190 std::stringstream its_path;
191 its_path << utility::get_base_path(configuration_) << std::hex << _client;
192 std::shared_ptr<local_client_endpoint_impl> its_endpoint;
193
194#ifdef _WIN32
195 boost::asio::ip::address address = boost::asio::ip::address::from_string("127.0.0.1");
196 int port = VSOMEIP_INTERNAL_BASE_PORT + _client;
197 VSOMEIP_INFO << "Connecting to ["
198 << std::hex << _client << "] at " << port;
199#else
200 VSOMEIP_INFO << "Client [" << std::hex << rm_->get_client() << "] is connecting to ["
201 << std::hex << _client << "] at " << its_path.str();
202#endif
203 its_endpoint = std::make_shared<local_client_endpoint_impl>(
204 shared_from_this(), rm_->shared_from_this(),
205#ifdef _WIN32
206 boost::asio::ip::tcp::endpoint(address, port)
207#else
208 boost::asio::local::stream_protocol::endpoint(its_path.str())
209#endif
210 , io_, configuration_);
211
212 // Messages sent to the VSOMEIP_ROUTING_CLIENT are meant to be routed to
213 // external devices. Therefore, its local endpoint must not be found by
214 // a call to find_local. Thus it must not be inserted to the list of local
215 // clients.
216 if (_client != VSOMEIP_ROUTING_CLIENT) {
217 local_endpoints_[_client] = its_endpoint;
218 }
219 rm_->register_client_error_handler(_client, its_endpoint);
220
221 return its_endpoint;
222}

224std::shared_ptr<endpoint> endpoint_manager_base::find_local_unlocked(client_t _client) {
225 std::shared_ptr<endpoint> its_endpoint;
226 auto found_endpoint = local_endpoints_.find(_client);
227 if (found_endpoint != local_endpoints_.end()) {
228 its_endpoint = found_endpoint->second;
229 }
230 return (its_endpoint);
231}
