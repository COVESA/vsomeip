137 basic_socket_acceptor_ext(boost::asio::io_service& io_service,
138 const endpoint_type& endpoint, bool reuse_addr = true)
139 : basic_io_object<SocketAcceptorService>(io_service)
140 {
141 boost::system::error_code ec;
142 const protocol_type protocol = endpoint.protocol();
143 this->get_service().open(this->get_implementation(), protocol, ec);
144 boost::asio::detail::throw_error(ec, "open");
145 if (reuse_addr)
146 {
147 this->get_service().set_option(this->get_implementation(),
148 socket_base::reuse_address(true), ec);
149 boost::asio::detail::throw_error(ec, "set_option");
150 }
151 this->get_service().bind(this->get_implementation(), endpoint, ec);
152 boost::asio::detail::throw_error(ec, "bind");
153 this->get_service().listen(this->get_implementation(),
154 socket_base::max_connections, ec);
155 boost::asio::detail::throw_error(ec, "listen");
156 }
