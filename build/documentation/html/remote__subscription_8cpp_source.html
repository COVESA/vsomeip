26bool
27remote_subscription::operator==(
28 const remote_subscription &_other) const {
29 auto own_egi = eventgroupinfo_.lock();
30 auto other_egi = _other.eventgroupinfo_.lock();
31 bool reliable_equal(true);
32 if (reliable_ && _other.reliable_) {
33 reliable_equal = (reliable_ == _other.reliable_);
34 }
35 bool unreliable_equal(true);
36 if (unreliable_ && _other.unreliable_) {
37 unreliable_equal = (unreliable_ == _other.unreliable_);
38 }
39 return (own_egi && other_egi && own_egi == other_egi && unreliable_equal
40 && reliable_equal);
41}

171void
172remote_subscription::set_client_state(const client_t _client,
173 remote_subscription_state_e _state) {
174 std::lock_guard<std::mutex> its_lock(mutex_);
175 auto found_item = clients_.find(_client);
176 if (found_item != clients_.end()) {
177 found_item->second.first = _state;
178 if (found_item->second.second == std::chrono::steady_clock::time_point()
179 && (_state == remote_subscription_state_e::SUBSCRIPTION_ACKED
180 || _state == remote_subscription_state_e::SUBSCRIPTION_NACKED)) {
181 found_item->second.second = std::chrono::steady_clock::now()
182 + std::chrono::seconds(ttl_);
183 }
184 }
185}

261std::set<client_t>
262remote_subscription::update(const std::set<client_t> &_clients,
263 const std::chrono::steady_clock::time_point &_timepoint,
264 const bool _is_subscribe) {
265 std::set<client_t> its_changed;
266
267 std::lock_guard<std::mutex> its_lock(mutex_);
268 for (const auto &its_client : _clients) {
269 auto found_client = clients_.find(its_client);
270 if (_is_subscribe) {
271 if (found_client != clients_.end()) {
272 found_client->second.second = _timepoint;
273 } else {
274 its_changed.insert(its_client);
275 }
276 } else {
277 if (found_client != clients_.end()) {
278 its_changed.insert(its_client);
279 }
280 }
281 }
282
283 for (const auto &its_client : its_changed) {
284 if (_is_subscribe) {
285 clients_[its_client] = std::make_pair(
286 remote_subscription_state_e::SUBSCRIPTION_PENDING, _timepoint);
287 } else {
288 clients_.erase(its_client);
289 }
290 }
291
292 return its_changed;
293}
