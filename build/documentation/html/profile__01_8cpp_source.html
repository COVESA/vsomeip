17uint8_t profile_01::compute_crc(const profile_config &_config, const e2e_buffer &_buffer) {
18 uint8_t computed_crc = 0xFF;
19 e2e_buffer data_id_buffer; //(_data, _data+_size);
20 data_id_buffer.push_back((uint8_t) (_config.data_id_ >> 8)); // insert MSB
21 data_id_buffer.push_back((uint8_t) _config.data_id_); // insert LSB
22
23 switch (_config.data_id_mode_) {
24 case p01_data_id_mode::E2E_P01_DATAID_BOTH: // CRC over 2 bytes
25 /*
26 * Two bytes are included in the CRC (double ID configuration) This is used in E2E variant 1A.
27 */
28 // CRC = Crc_CalculateCRC8(Config->DataID, 1, 0xFF, FALSE)
29 computed_crc = e2e_crc::calculate_profile_01(buffer_view(data_id_buffer, 1, 2), 0xFF); //CRC over low byte of Data ID (LSB)
30
31 // CRC = Crc_CalculateCRC8(Config->DataID >> 8, 1, CRC, FALSE)
32 computed_crc = e2e_crc::calculate_profile_01(buffer_view(data_id_buffer, 0, 1), computed_crc); //CRC over high byte of Data ID (MSB)
33
34 break;
35 case p01_data_id_mode::E2E_P01_DATAID_LOW: // CRC over low byte only
36 /*
37 * Only the low byte is included, the high byte is never used.
38 * This is applicable if the IDs in a particular system are 8 bits
39 */
40 // CRC = Crc_CalculateCRC8(Config->DataID, 1, 0xFF, FALSE)
41 computed_crc = e2e_crc::calculate_profile_01(buffer_view(data_id_buffer, 1, 2), 0xFF); //CRC over low byte of Data ID (LSB)
42 break;
43
44 case p01_data_id_mode::E2E_P01_DATAID_ALT:
45 /* One of the two bytes byte is included, alternating high and low byte,
46 * depending on parity of the counter (alternating ID configuration).
47 * For an even counter, the low byte is included.
48 * For an odd counter, the high byte is included.
49 * This is used in E2E variant 1B.
50 *
51 if( counter % 2 == 0) {
52 // CRC = Crc_CalculateCRC8(Config->DataID, 1, 0xFF, FALSE)
53 computed_crc = crc::e2e_crc::calculate_profile_01(buffer::buffer_view(data_id_buffer, 1, 2), 0xFF); //CRC over low byte of Data ID (LSB)
54 } else {
55 // CRC = Crc_CalculateCRC8(Config->DataID >> 8, 1, 0xFF, FALSE)
56 computed_crc = crc::e2e_crc::calculate_profile_01(buffer::buffer_view(data_id_buffer, 0, 1), 0xFF); //CRC over high byte of Data ID (MSB)
57 }
58 */
59 break;
60
61 case p01_data_id_mode::E2E_P01_DATAID_NIBBLE:
62 /*
63 * The low byte is included in the implicit CRC calculation,
64 * the low nibble of the high byte is transmitted along with
65 * the data (i.e. it is explicitly included), the high nibble of
66 * the high byte is not used. This is applicable for the IDs
67 * up to 12 bits. This is used in E2E variant 1C.
68 */
69 // CRC = Crc_CalculateCRC8(Config->DataID, 1, 0xFF, FALSE)
70 computed_crc = e2e_crc::calculate_profile_01(buffer_view(data_id_buffer, 1, 2), 0xFF); //CRC over low byte of Data ID (LSB)
71
72 // CRC = Crc_CalculateCRC8 (0, 1, CRC, FALSE)
73 data_id_buffer.clear();
74 data_id_buffer.push_back(0x00);
75 computed_crc = e2e_crc::calculate_profile_01(buffer_view(data_id_buffer, 0, 1), computed_crc); // CRC with 0x00
76 break;
77
78 default:
79 break;
80 }
81
82 // Compute CRC over the area before the CRC (if CRC is not the first byte)
83 if (_config.crc_offset_ >= 1) {
84 // CRC = Crc_CalculateCRC8 (Data, (Config->CRCOffset / 8), CRC, FALSE)
85 computed_crc = e2e_crc::calculate_profile_01(buffer_view(_buffer, 0, _config.crc_offset_), computed_crc);
86 }
87
88 // Compute the area after CRC, if CRC is not the last byte. Start with the byte after CRC, finish with the last byte of Data.
89 if ((_config.crc_offset_) < (_config.data_length_ / 8) - 1) {
90 // CRC = Crc_CalculateCRC8 (& Data[Config->CRCOffset/8 + 1], (Config->DataLength / 8 - Config->CRCOffset / 8 - 1), CRC, FALSE)
91 computed_crc = e2e_crc::calculate_profile_01(buffer_view(_buffer, static_cast<size_t>(_config.crc_offset_ + 1), _buffer.size()), computed_crc);
92 }
93
94 // CRC = CRC ^ 0xFF
95 // To negate the last XOR 0xFF operation done on computed CRC by the last CalculateCRC8(), there is a XORing doneexternally by E2E Library
96 computed_crc = computed_crc ^ 0xFFU;
97 return computed_crc;
98}
