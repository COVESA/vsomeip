102bool event::set_payload_dont_notify(const std::shared_ptr<payload> &_payload) {
103 std::lock_guard<std::mutex> its_lock(mutex_);
104 if (is_cache_placeholder_) {

117void event::set_payload(const std::shared_ptr<payload> &_payload, bool _force) {
118 std::lock_guard<std::mutex> its_lock(mutex_);
119 if (is_provided_) {
120 if (set_payload_helper(_payload, _force)) {
121 reset_payload(_payload);
122 if (is_updating_on_change_) {
123 if (change_resets_cycle_)
124 stop_cycle();
125
126 notify();
127
128 if (change_resets_cycle_)
129 start_cycle();
130 }
131 }
132 } else {
133 VSOMEIP_INFO << "Can't set payload for event " << std::hex
134 << message_->get_method() << " as it isn't provided";
135 }
136}

172bool event::set_payload_notify_pending(const std::shared_ptr<payload> &_payload) {
173 std::lock_guard<std::mutex> its_lock(mutex_);
174 if (!is_set_ && is_provided_) {
175 reset_payload(_payload);
176
177 // Send pending initial events.
178 for (const auto &its_target : pending_) {
179 message_->set_session(routing_->get_session());
180 routing_->send_to(VSOMEIP_ROUTING_CLIENT,
181 its_target, message_);
182 }
183 pending_.clear();
184
185 return true;
186 }
187
188 return false;

359void event::add_ref(client_t _client, bool _is_provided) {
360 std::lock_guard<std::mutex> its_lock(refs_mutex_);
361 auto its_client = refs_.find(_client);
362 if (its_client == refs_.end()) {
363 refs_[_client][_is_provided] = 1;
364 } else {
365 auto its_provided = its_client->second.find(_is_provided);
366 if (its_provided == its_client->second.end()) {
367 refs_[_client][_is_provided] = 1;
368 } else {
369 its_provided->second++;
370 }
371 }
372}

374void event::remove_ref(client_t _client, bool _is_provided) {
375 std::lock_guard<std::mutex> its_lock(refs_mutex_);
376 auto its_client = refs_.find(_client);
377 if (its_client != refs_.end()) {
378 auto its_provided = its_client->second.find(_is_provided);
379 if (its_provided != its_client->second.end()) {
380 its_provided->second--;
381 if (0 == its_provided->second) {
382 its_client->second.erase(_is_provided);
383 if (0 == its_client->second.size()) {
384 refs_.erase(_client);
385 }
386 }
387 }
388 }

397 std::lock_guard<std::mutex> its_lock(eventgroups_mutex_);
398 bool ret = false;
399 if (_force // remote events managed by rm_impl
400 || is_provided_ // events provided by rm_proxies
401 || is_shadow_ // local events managed by rm_impl
402 || is_cache_placeholder_) {
403 ret = eventgroups_[_eventgroup].insert(_client).second;
404 } else {
405 VSOMEIP_WARNING << __func__ << ": Didnt' insert client "
406 << std::hex << std::setw(4) << std::setfill('0') << _client
407 << " to eventgroup 0x"
408 << std::hex << std::setw(4) << std::setfill('0') << _eventgroup;
409 }
410 return ret;

495bool event::compare(const std::shared_ptr<payload> &_lhs,
496 const std::shared_ptr<payload> &_rhs) const {
497 bool is_change = (_lhs->get_length() != _rhs->get_length());
498 if (!is_change) {
499 std::size_t its_pos = 0;
500 const byte_t *its_old_data = _lhs->get_data();
501 const byte_t *its_new_data = _rhs->get_data();
502 while (!is_change && its_pos < _lhs->get_length()) {
503 is_change = (*its_old_data++ != *its_new_data++);
504 its_pos++;
505 }
506 }
507 return is_change;
508}
