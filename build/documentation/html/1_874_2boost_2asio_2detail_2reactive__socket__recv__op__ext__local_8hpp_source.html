52 static status do_perform(reactor_op* base)
53 {
54 reactive_socket_recv_op_base_ext_local* o(
55 static_cast<reactive_socket_recv_op_base_ext_local*>(base));
56
57 typedef buffer_sequence_adapter<boost::asio::mutable_buffer,
58 MutableBufferSequence> bufs_type;
59
60 status result;
61 if (bufs_type::is_single_buffer)
62 {
63 result = socket_ops::non_blocking_recv1(o->socket_,
64 bufs_type::first(o->buffers_).data(),
65 bufs_type::first(o->buffers_).size(), o->flags_,
66 (o->state_ & socket_ops::stream_oriented) != 0,
67 o->ec_, o->bytes_transferred_) ? done : not_done;
68 }
69 else
70 {
71 bufs_type bufs(o->buffers_);
72 result = socket_ops::non_blocking_recv(o->socket_,
73 bufs.buffers(), bufs.count(), o->flags_,
74 (o->state_ & socket_ops::stream_oriented) != 0,
75 o->ec_, o->bytes_transferred_) ? done : not_done;
76 }
77
78 if (result == done)
79 if ((o->state_ & socket_ops::stream_oriented) != 0)
80 if (o->bytes_transferred_ == 0)
81 result = done_and_exhausted;
82
83 BOOST_ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_recv",
84 o->ec_, o->bytes_transferred_));
85
86 return result;
87 }
88
89private:
90 socket_type socket_;
91 socket_ops::state_type state_;
92 MutableBufferSequence buffers_;
93 socket_base::message_flags flags_;
94};

115 static void do_complete(void* owner, operation* base,
116 const boost::system::error_code& /*ec*/,
117 std::size_t /*bytes_transferred*/)
118 {
119 // Take ownership of the handler object.
120 reactive_socket_recv_op_ext_local* o(static_cast<reactive_socket_recv_op_ext_local*>(base));
121 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
122
123 BOOST_ASIO_HANDLER_COMPLETION((*o));
124
125 // Take ownership of the operation's outstanding work.
126 handler_work<Handler, IoExecutor> w(
127 BOOST_ASIO_MOVE_CAST2(handler_work<Handler, IoExecutor>)(
128 o->work_));
129
130 // Make a copy of the handler so that the memory can be deallocated before
131 // the upcall is made. Even if we're not about to make an upcall, a
132 // sub-object of the handler may be the true owner of the memory associated
133 // with the handler. Consequently, a local copy of the handler is required
134 // to ensure that any owning sub-object remains valid until after we have
135 // deallocated the memory here.
136 detail::binder4<Handler, boost::system::error_code, std::size_t, std::uint32_t, std::uint32_t>
137 handler(o->handler_, o->ec_, o->bytes_transferred_, o->uid_, o->gid_);
138 p.h = boost::asio::detail::addressof(handler.handler_);
139 p.reset();
140
141 // Make the upcall if required.
142 if (owner)
143 {
144 fenced_block b(fenced_block::half);
145 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_, handler.arg3, handler.arg4));
146 w.complete(handler, handler.handler_);
147 BOOST_ASIO_HANDLER_INVOCATION_END;
148 }
149 }
150
151private:
152 Handler handler_;
153 handler_work<Handler, IoExecutor> work_;
154};
