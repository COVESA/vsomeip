275 bool operator<(const subscription_data_t &_other) const {
276 return (service_ < _other.service_
277 || (service_ == _other.service_
278 && instance_ < _other.instance_)
279 || (service_ == _other.service_
280 && instance_ == _other.instance_
281 && eventgroup_ < _other.eventgroup_)
282 || (service_ == _other.service_
283 && instance_ == _other.instance_
284 && eventgroup_ == _other.eventgroup_

286 }
