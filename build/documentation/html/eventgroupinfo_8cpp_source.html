125void eventgroupinfo::add_event(const std::shared_ptr<event>& _event) {
126 std::lock_guard<std::mutex> its_lock(events_mutex_);
127 events_.insert(_event);
128
129 if (!reliability_auto_mode_ &&
130 _event->get_reliability() == reliability_type_e::RT_UNKNOWN) {
131 reliability_auto_mode_ = true;
132 return;
133 }
134
135 switch (_event->get_reliability()) {
136 case reliability_type_e::RT_RELIABLE:
137 if (reliability_ == reliability_type_e::RT_UNRELIABLE) {
138 reliability_ = reliability_type_e::RT_BOTH;
139 } else if (reliability_ != reliability_type_e::RT_BOTH) {
140 reliability_ = reliability_type_e::RT_RELIABLE;
141 }
142 break;
143 case reliability_type_e::RT_UNRELIABLE:
144 if (reliability_ == reliability_type_e::RT_RELIABLE) {
145 reliability_ = reliability_type_e::RT_BOTH;
146 } else if (reliability_ != reliability_type_e::RT_BOTH) {
147 reliability_ = reliability_type_e::RT_UNRELIABLE;
148 }
149 break;
150 case reliability_type_e::RT_BOTH:
151 reliability_ = reliability_type_e::RT_BOTH;
152 break;
153 default:
154 ;
155 }
156}

175uint32_t
176eventgroupinfo::get_unreliable_target_count() const {
177 uint32_t its_count(0);
178
179 std::lock_guard<std::mutex> its_lock(subscriptions_mutex_);
180 for (const auto &s : subscriptions_) {
181 auto its_subscription = s.second;
182 if (!its_subscription->get_parent()
183 && its_subscription->get_unreliable()) {
184 its_count++;
185 }
186 }
187
188 return its_count;

210bool
211eventgroupinfo::update_remote_subscription(
212 const std::shared_ptr<remote_subscription> &_subscription,
213 const std::chrono::steady_clock::time_point &_expiration,
214 std::set<client_t> &_changed, remote_subscription_id_t &_id,
215 const bool _is_subscribe) {
216 std::lock_guard<std::mutex> its_lock(subscriptions_mutex_);
217
218 for (const auto& its_item : subscriptions_) {
219 if (its_item.second->equals(_subscription)) {
220 // update existing subscription
221 _changed = its_item.second->update(
222 _subscription->get_clients(), _expiration, _is_subscribe);
223 _id = its_item.second->get_id();
224
225 // Copy acknowledgment states from existing subscription
226 for (const auto its_client : _subscription->get_clients()) {
227 _subscription->set_client_state(its_client,
228 its_item.second->get_client_state(its_client));
229 }
230
231 if (_is_subscribe) {
232 if (!_changed.empty()) {
233 // New clients:
234 // Let this be a child subscription
235 _subscription->set_parent(its_item.second);
236 update_id();
237 _subscription->set_id(id_);
238 subscriptions_[id_] = _subscription;
239 } else {
240 if (!_subscription->is_pending()) {
241 if (!_subscription->force_initial_events()) {
242 _subscription->set_initial(false);
243 }
244 } else {
245 its_item.second->set_answers(
246 its_item.second->get_answers() + 1);
247 _subscription->set_parent(its_item.second);
248 _subscription->set_answers(0);
249 }
250 }
251 } else {
252 if (its_item.second->is_pending()) {
253 for (const auto &its_event : events_)
254 its_event->remove_pending(
255 its_item.second->get_subscriber());
256 }
257 }
258
259 return true;
260 }
261 }
262
263 return false;
264}

343void
344eventgroupinfo::send_initial_events(
345 const std::shared_ptr<endpoint_definition> &_reliable,
346 const std::shared_ptr<endpoint_definition> &_unreliable) const {
347
348 std::set<std::shared_ptr<event> > its_reliable_events, its_unreliable_events;
349
350 // Build sets of reliable/unreliable events first to avoid having to
351 // hold the "events_mutex_" in parallel to the internal event mutexes.
352 {
353 std::lock_guard<std::mutex> its_lock(events_mutex_);
354 for (const auto &its_event : events_) {
355 if (its_event && its_event->get_type() == event_type_e::ET_FIELD) {
356 auto its_reliability = its_event->get_reliability();
357#ifdef VSOMEIP_ENABLE_COMPAT
358 if (its_reliability == reliability_type_e::RT_UNKNOWN) {
359 if (_reliable) {
360 if (_unreliable) {
361 its_reliability = reliability_type_e::RT_BOTH;
362 } else {
363 its_reliability = reliability_type_e::RT_RELIABLE;
364 }
365 } else if (_unreliable) {
366 its_reliability = reliability_type_e::RT_UNRELIABLE;
367 }
368 }
369#endif
370 switch (its_reliability) {
371 case reliability_type_e::RT_RELIABLE:
372 its_reliable_events.insert(its_event);
373 break;
374 case reliability_type_e::RT_UNRELIABLE:
375 its_unreliable_events.insert(its_event);
376 break;
377 case reliability_type_e::RT_BOTH:
378 its_reliable_events.insert(its_event);
379 its_unreliable_events.insert(its_event);
380 break;
381 default:
382 VSOMEIP_WARNING << __func__ << "Event reliability unknown: ["
383 << std::hex << std::setw(4) << std::setfill('0') << service_ << "."
384 << std::hex << std::setw(4) << std::setfill('0') << instance_ << "."
385 << std::hex << std::setw(4) << std::setfill('0') << eventgroup_ << "."
386 << std::hex << std::setw(4) << std::setfill('0') << its_event->get_event() << "]";
387 }
388 }
389 }
390 }
391
392 // Send events
393 for (const auto its_event : its_reliable_events)
394 its_event->notify_one(VSOMEIP_ROUTING_CLIENT, _reliable);
395
396 for (const auto its_event : its_unreliable_events)
397 its_event->notify_one(VSOMEIP_ROUTING_CLIENT, _unreliable);
398}
