53void udp_client_endpoint_impl::connect() {
54 std::lock_guard<std::mutex> its_lock(socket_mutex_);
55 boost::system::error_code its_error;
56 socket_->open(remote_.protocol(), its_error);
57 if (!its_error || its_error == boost::asio::error::already_open) {
58 // Enable SO_REUSEADDR to avoid bind problems with services going offline
59 // and coming online again and the user has specified only a small number
60 // of ports in the clients section for one service instance
61 socket_->set_option(boost::asio::socket_base::reuse_address(true), its_error);
62 if (its_error) {
63 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't enable "
64 << "SO_REUSEADDR: " << its_error.message() << " remote:"
65 << get_address_port_remote();
66 }
67 socket_->set_option(boost::asio::socket_base::receive_buffer_size(
68 udp_receive_buffer_size_), its_error);
69 if (its_error) {
70 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't set "
71 << "SO_RCVBUF: " << its_error.message() << " to: "
72 << std::dec << udp_receive_buffer_size_ << " remote:"
73 << get_address_port_remote();
74 } else {
75 boost::asio::socket_base::receive_buffer_size its_option;
76 socket_->get_option(its_option, its_error);
77 if (its_error) {
78 VSOMEIP_WARNING << "udp_client_endpoint_impl::connect: couldn't get "
79 << "SO_RCVBUF: " << its_error.message() << " remote:"
80 << get_address_port_remote();
81 } else {
82 VSOMEIP_INFO << "udp_client_endpoint_impl::connect: SO_RCVBUF is: "
83 << std::dec << its_option.value();
84 }
85 }
86
87#ifndef _WIN32
88 // If specified, bind to device
89 std::string its_device(configuration_->get_device());
90 if (its_device != "") {
91 if (setsockopt(socket_->native_handle(),
92 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), (int)its_device.size()) == -1) {
93 VSOMEIP_WARNING << "UDP Client: Could not bind to device \"" << its_device << "\"";
94 }
95 }
96#endif
97
98 // Bind address and, optionally, port.
99 boost::system::error_code its_bind_error;
100 socket_->bind(local_, its_bind_error);
101 if(its_bind_error) {
102 VSOMEIP_WARNING << "udp_client_endpoint::connect: "
103 "Error binding socket: " << its_bind_error.message()
104 << " remote:" << get_address_port_remote();
105 try {
106 // don't connect on bind error to avoid using a random port
107 strand_.post(std::bind(&client_endpoint_impl::connect_cbk,
108 shared_from_this(), its_bind_error));
109 } catch (const std::exception &e) {
110 VSOMEIP_ERROR << "udp_client_endpoint_impl::connect: "
111 << e.what() << " remote:" << get_address_port_remote();
112 }
113 return;
114 }
115
116 state_ = cei_state_e::CONNECTING;
117 socket_->async_connect(
118 remote_,
119 strand_.wrap(
120 std::bind(
121 &udp_client_endpoint_base_impl::connect_cbk,
122 shared_from_this(),
123 std::placeholders::_1
124 )
125 )
126 );
127 } else {
128 VSOMEIP_WARNING << "udp_client_endpoint::connect: Error opening socket: "
129 << its_error.message() << " remote:" << get_address_port_remote();
130 strand_.post(std::bind(&udp_client_endpoint_base_impl::connect_cbk,
131 shared_from_this(), its_error));
132 }
133}

139void udp_client_endpoint_impl::restart(bool _force) {
140 if (!_force && state_ == cei_state_e::CONNECTING) {
141 return;
142 }
143 state_ = cei_state_e::CONNECTING;
144 {
145 std::lock_guard<std::mutex> its_lock(mutex_);
146 queue_.clear();
147 }
148 std::string local;
149 {
150 std::lock_guard<std::mutex> its_lock(socket_mutex_);
151 local = get_address_port_local();
152 }
153 shutdown_and_close_socket(false);
154 was_not_connected_ = true;
155 reconnect_counter_ = 0;
156 VSOMEIP_WARNING << "uce::restart: local: " << local
157 << " remote: " << get_address_port_remote();
158 start_connect_timer();
159}

251void udp_client_endpoint_impl::receive_cbk(
252 boost::system::error_code const &_error, std::size_t _bytes,
253 const message_buffer_ptr_t& _recv_buffer) {
254 if (_error == boost::asio::error::operation_aborted) {
255 // endpoint was stopped
256 return;
257 }
258 std::shared_ptr<routing_host> its_host = routing_host_.lock();
259 if (!_error && 0 < _bytes && its_host) {
260#if 0
261 std::stringstream msg;
262 msg << "ucei::rcb(" << _error.message() << "): ";
263 for (std::size_t i = 0; i < _bytes; ++i)
264 msg << std::hex << std::setw(2) << std::setfill('0')
265 << (int) (*_recv_buffer)[i] << " ";
266 VSOMEIP_INFO << msg.str();
267#endif
268 std::size_t remaining_bytes = _bytes;
269 std::size_t i = 0;
270
271 do {
272 uint64_t read_message_size
273 = utility::get_message_size(&(*_recv_buffer)[i],
274 remaining_bytes);
275 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
276 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
277 return;
278 }
279 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
280 if (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
281 current_message_size <= remaining_bytes) {
282 if (remaining_bytes - current_message_size > remaining_bytes) {
283 VSOMEIP_ERROR << "buffer underflow in udp client endpoint ~> abort!";
284 return;
285 } else if (current_message_size > VSOMEIP_RETURN_CODE_POS &&
286 ((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION ||
287 !utility::is_valid_message_type(tp::tp::tp_flag_unset((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])) ||
288 !utility::is_valid_return_code(static_cast<return_code_e>((*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS]))
289 )) {
290 if ((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
291 VSOMEIP_ERROR << "uce: Wrong protocol version: 0x"
292 << std::hex << std::setw(2) << std::setfill('0')
293 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_PROTOCOL_VERSION_POS])
294 << " local: " << get_address_port_local()
295 << " remote: " << get_address_port_remote();
296 // ensure to send back a message w/ wrong protocol version
297 its_host->on_message(&(*_recv_buffer)[i],
298 VSOMEIP_SOMEIP_HEADER_SIZE + 8, this,
299 boost::asio::ip::address(),
300 VSOMEIP_ROUTING_CLIENT,
301 std::make_pair(ANY_UID, ANY_GID),
302 remote_address_,
303 remote_port_);
304 } else if (!utility::is_valid_message_type(tp::tp::tp_flag_unset(
305 (*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS]))) {
306 VSOMEIP_ERROR << "uce: Invalid message type: 0x"
307 << std::hex << std::setw(2) << std::setfill('0')
308 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])
309 << " local: " << get_address_port_local()
310 << " remote: " << get_address_port_remote();
311 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
312 (*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS]))) {
313 VSOMEIP_ERROR << "uce: Invalid return code: 0x"
314 << std::hex << std::setw(2) << std::setfill('0')
315 << std::uint32_t((*_recv_buffer)[i + VSOMEIP_RETURN_CODE_POS])
316 << " local: " << get_address_port_local()
317 << " remote: " << get_address_port_remote();
318 }
319 receive();
320 return;
321 } else if (tp::tp::tp_flag_is_set((*_recv_buffer)[i + VSOMEIP_MESSAGE_TYPE_POS])) {
322 const auto res = tp_reassembler_->process_tp_message(
323 &(*_recv_buffer)[i], current_message_size,
324 remote_address_, remote_port_);
325 if (res.first) {
326 its_host->on_message(&res.second[0],
327 static_cast<std::uint32_t>(res.second.size()),
328 this, boost::asio::ip::address(),
329 VSOMEIP_ROUTING_CLIENT,
330 std::make_pair(ANY_UID, ANY_GID),
331 remote_address_,
332 remote_port_);
333 }
334 } else {
335 its_host->on_message(&(*_recv_buffer)[i], current_message_size,
336 this, boost::asio::ip::address(),
337 VSOMEIP_ROUTING_CLIENT,
338 std::make_pair(ANY_UID, ANY_GID),
339 remote_address_,
340 remote_port_);
341 }
342 remaining_bytes -= current_message_size;
343 } else {
344 VSOMEIP_ERROR << "Received a unreliable vSomeIP message with bad "
345 "length field. Message size: " << current_message_size
346 << " Bytes. From: " << remote_.address() << ":"
347 << remote_.port() << ". Dropping message.";
348 remaining_bytes = 0;
349 }
350 i += current_message_size;
351 } while (remaining_bytes > 0);
352 }
353 if (!_error) {
354 receive();
355 } else {
356 if (_error == boost::asio::error::connection_refused) {
357 shutdown_and_close_socket(false);
358 } else {
359 receive();
360 }
361 }

364const std::string udp_client_endpoint_impl::get_address_port_remote() const {
365 boost::system::error_code ec;
366 std::string its_address_port;
367 its_address_port.reserve(21);
368 boost::asio::ip::address its_address;
369 if (get_remote_address(its_address)) {
370 its_address_port += its_address.to_string();
371 }
372 its_address_port += ":";
373 its_address_port += std::to_string(remote_port_);
374 return its_address_port;
375}

380 boost::system::error_code ec;

383 if (!ec) {
384 its_address_port += its_local_endpoint.address().to_string(ec);
385 its_address_port += ":";
386 its_address_port.append(std::to_string(its_local_endpoint.port()));
387 }
388 }
389 return its_address_port;
