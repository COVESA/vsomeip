227void routing_manager_impl::stop() {
228 // Ensure to StopOffer all services that are offered by the application hosting the rm
229 local_services_map_t its_services;
230 {
231 std::lock_guard<std::mutex> its_lock(local_services_mutex_);
232 for (const auto& s : local_services_) {
233 for (const auto& i : s.second) {
234 if (std::get<2>(i.second) == client_) {
235 its_services[s.first][i.first] = i.second;
236 }
237 }
238 }
239
240 }
241 for (const auto& s : its_services) {
242 for (const auto& i : s.second) {
243 on_stop_offer_service(std::get<2>(i.second), s.first, i.first,
244 std::get<0>(i.second), std::get<1>(i.second));
245 }
246 }
247
248 {
249 std::lock_guard<std::mutex> its_lock(version_log_timer_mutex_);
250 version_log_timer_.cancel();
251 }
252#ifndef _WIN32
253 {
254 boost::system::error_code ec;
255 std::lock_guard<std::mutex> its_lock(memory_log_timer_mutex_);
256 memory_log_timer_.cancel(ec);
257 }
258 if (netlink_connector_) {
259 netlink_connector_->stop();
260 }
261#endif
262
263 {
264 std::lock_guard<std::mutex> its_lock(status_log_timer_mutex_);
265 boost::system::error_code ec;
266 status_log_timer_.cancel(ec);
267 }
268
269 {
270 std::lock_guard<std::mutex> its_lock(statistics_log_timer_mutex_);
271 boost::system::error_code ec;
272 statistics_log_timer_.cancel(ec);
273 }
274
275 host_->on_state(state_type_e::ST_DEREGISTERED);
276
277 if (discovery_)
278 discovery_->stop();
279 stub_->stop();
280
281 for (const auto client : ep_mgr_->get_connected_clients()) {
282 if (client != VSOMEIP_ROUTING_CLIENT) {
283 remove_local(client, true);
284 }
285 }

288bool routing_manager_impl::insert_offer_command(service_t _service, instance_t _instance, uint8_t _command,
289 client_t _client, major_version_t _major, minor_version_t _minor) {
290 std::lock_guard<std::mutex> its_lock(offer_serialization_mutex_);
291 // flag to indicate whether caller of this function can start directly processing the command
292 bool must_process(false);
293 auto found_service_instance = offer_commands_.find(std::make_pair(_service, _instance));
294 if (found_service_instance != offer_commands_.end()) {
295 // if nothing is queued
296 if (found_service_instance->second.empty()) {
297 must_process = true;
298 }
299 found_service_instance->second.push_back(
300 std::make_tuple(_command, _client, _major, _minor));
301 } else {
302 // nothing is queued -> add command to queue and process command directly
303 offer_commands_[std::make_pair(_service, _instance)].push_back(
304 std::make_tuple(_command, _client, _major, _minor));
305 must_process = true;
306 }
307 return must_process;
308}
309
310bool routing_manager_impl::erase_offer_command(service_t _service, instance_t _instance) {
311 std::lock_guard<std::mutex> its_lock(offer_serialization_mutex_);
312 auto found_service_instance = offer_commands_.find(std::make_pair(_service, _instance));
313 if (found_service_instance != offer_commands_.end()) {
314 // erase processed command
315 if (!found_service_instance->second.empty()) {
316 found_service_instance->second.pop_front();
317 if (!found_service_instance->second.empty()) {
318 // check for other commands to be processed
319 auto its_command = found_service_instance->second.front();
320 if (std::get<0>(its_command) == VSOMEIP_OFFER_SERVICE) {
321 io_.post([&, its_command, _service, _instance](){
322 offer_service(std::get<1>(its_command), _service, _instance,
323 std::get<2>(its_command), std::get<3>(its_command), false);
324 });
325 } else {
326 io_.post([&, its_command, _service, _instance](){
327 stop_offer_service(std::get<1>(its_command), _service, _instance,
328 std::get<2>(its_command), std::get<3>(its_command), false);
329 });
330 }
331 }
332 }
333 }
334 return true;
335}

344bool routing_manager_impl::offer_service(client_t _client,
345 service_t _service, instance_t _instance,
346 major_version_t _major, minor_version_t _minor,
347 bool _must_queue) {
348
349 VSOMEIP_INFO << "OFFER("
350 << std::hex << std::setw(4) << std::setfill('0') << _client <<"): ["
351 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
352 << std::hex << std::setw(4) << std::setfill('0') << _instance
353 << ":" << std::dec << int(_major) << "." << std::dec << _minor << "]"
354 << " (" << std::boolalpha << _must_queue << ")";
355
356 // only queue commands if method was NOT called via erase_offer_command()
357 if (_must_queue) {
358 if (!insert_offer_command(_service, _instance, VSOMEIP_OFFER_SERVICE,
359 _client, _major, _minor)) {
360 return false;
361 }
362 }
363
364 // Check if the application hosted by routing manager is allowed to offer
365 // offer_service requests of local proxies are checked in rms::on:message
366 if (_client == get_client()) {
367#ifdef _WIN32
368 std::uint32_t its_routing_uid = ANY_UID;
369 std::uint32_t its_routing_gid = ANY_GID;
370#else
371 std::uint32_t its_routing_uid = getuid();
372 std::uint32_t its_routing_gid = getgid();
373#endif
374 if (!security::get()->is_offer_allowed(its_routing_uid, its_routing_gid,
375 _client, _service, _instance)) {
376 VSOMEIP_WARNING << "routing_manager_impl::offer_service: "
377 << std::hex << "Security: Client 0x" << _client
378 << " isn't allowed to offer the following service/instance "
379 << _service << "/" << _instance
380 << " ~> Skip offer!";
381 erase_offer_command(_service, _instance);
382 return false;
383 }
384 }
385
386 if (!handle_local_offer_service(_client, _service, _instance, _major, _minor)) {
387 erase_offer_command(_service, _instance);
388 return false;
389 }
390
391 {
392 std::lock_guard<std::mutex> its_lock(pending_sd_offers_mutex_);
393 if (if_state_running_) {
394 init_service_info(_service, _instance, true);
395 } else {
396 pending_sd_offers_.push_back(std::make_pair(_service, _instance));
397 }
398 }
399
400 if (discovery_) {
401 std::shared_ptr<serviceinfo> its_info = find_service(_service, _instance);
402 if (its_info) {
403 discovery_->offer_service(its_info);
404 }
405 }
406
407 {
408 std::lock_guard<std::mutex> ist_lock(pending_subscription_mutex_);
409 std::set<event_t> its_already_subscribed_events;
410 for (auto &ps : pending_subscriptions_) {
411 if (ps.service_ == _service
412 && ps.instance_ == _instance
413 && ps.major_ == _major) {
414 insert_subscription(ps.service_, ps.instance_,
415 ps.eventgroup_, ps.event_, client_, &its_already_subscribed_events);
416#if 0
417 VSOMEIP_ERROR << __func__
418 << ": event="
419 << std::hex << ps.service_ << "."
420 << std::hex << ps.instance_ << "."
421 << std::hex << ps.event_;
422#endif
423 }
424 }
425
426 send_pending_subscriptions(_service, _instance, _major);
427 }
428 stub_->on_offer_service(_client, _service, _instance, _major, _minor);
429 on_availability(_service, _instance, true, _major, _minor);
430 erase_offer_command(_service, _instance);
431 return true;
432}

441void routing_manager_impl::stop_offer_service(client_t _client,
442 service_t _service, instance_t _instance,
443 major_version_t _major, minor_version_t _minor,
444 bool _must_queue) {
445
446 VSOMEIP_INFO << "STOP OFFER("
447 << std::hex << std::setw(4) << std::setfill('0') << _client <<"): ["
448 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
449 << std::hex << std::setw(4) << std::setfill('0') << _instance
450 << ":" << std::dec << int(_major) << "." << _minor << "]"
451 << " (" << std::boolalpha << _must_queue << ")";
452
453 if (_must_queue) {
454 if (!insert_offer_command(_service, _instance, VSOMEIP_STOP_OFFER_SERVICE,
455 _client, _major, _minor)) {
456 return;
457 }
458 }
459
460 bool is_local(false);
461 {
462 std::shared_ptr<serviceinfo> its_info = find_service(_service, _instance);
463 is_local = (its_info && its_info->is_local());
464 }
465 if (is_local) {
466 {
467 std::lock_guard<std::mutex> its_lock(pending_sd_offers_mutex_);
468 for (auto it = pending_sd_offers_.begin(); it != pending_sd_offers_.end(); ) {
469 if (it->first == _service && it->second == _instance) {
470 it = pending_sd_offers_.erase(it);
471 break;
472 } else {
473 ++it;
474 }
475 }
476 }
477
478 on_stop_offer_service(_client, _service, _instance, _major, _minor);
479 stub_->on_stop_offer_service(_client, _service, _instance, _major, _minor);
480 on_availability(_service, _instance, false, _major, _minor);
481 } else {
482 VSOMEIP_WARNING << __func__ << " received STOP_OFFER("
483 << std::hex << std::setw(4) << std::setfill('0') << _client <<"): ["
484 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
485 << std::hex << std::setw(4) << std::setfill('0') << _instance
486 << ":" << std::dec << int(_major) << "." << _minor << "] "
487 << "for remote service --> ignore";
488 erase_offer_command(_service, _instance);
489 }
490}

492void routing_manager_impl::request_service(client_t _client, service_t _service,
493 instance_t _instance, major_version_t _major, minor_version_t _minor) {
494
495 VSOMEIP_INFO << "REQUEST("
496 << std::hex << std::setw(4) << std::setfill('0') << _client << "): ["
497 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
498 << std::hex << std::setw(4) << std::setfill('0') << _instance << ":"
499 << std::dec << int(_major) << "." << std::dec << _minor << "]";
500
501 routing_manager_base::request_service(_client,
502 _service, _instance, _major, _minor);
503
504 auto its_info = find_service(_service, _instance);
505 if (!its_info) {
506 requested_service_add(_client, _service, _instance, _major, _minor);
507 if (discovery_) {
508 if (!configuration_->is_local_service(_service, _instance)) {
509 // Non local service instance ~> tell SD to find it!
510 discovery_->request_service(_service, _instance, _major, _minor,
511 DEFAULT_TTL);
512 } else {
513 VSOMEIP_INFO << std::hex
514 << "Avoid trigger SD find-service message"
515 << " for local service/instance/major/minor: "
516 << _service << "/" << _instance << std::dec
517 << "/" << (uint32_t)_major << "/" << _minor;
518 }
519 }
520 } else {
521 if ((_major == its_info->get_major()
522 || DEFAULT_MAJOR == its_info->get_major()
523 || ANY_MAJOR == _major)
524 && (_minor <= its_info->get_minor()
525 || DEFAULT_MINOR == its_info->get_minor()
526 || _minor == ANY_MINOR)) {
527 if(!its_info->is_local()) {
528 requested_service_add(_client, _service, _instance, _major, _minor);
529 if (discovery_) {
530 // Non local service instance ~> tell SD to find it!
531 discovery_->request_service(_service, _instance, _major,
532 _minor, DEFAULT_TTL);
533 }
534 its_info->add_client(_client);
535 ep_mgr_impl_->find_or_create_remote_client(
536 _service, _instance, true);
537 }
538 }
539 }
540
541 if (_client == get_client()) {
542 stub_->create_local_receiver();
543
544 service_data_t request = { _service, _instance, _major, _minor };
545 std::set<service_data_t> requests;
546 requests.insert(request);
547 stub_->handle_requests(_client, requests);
548 }
549}

551void routing_manager_impl::release_service(client_t _client, service_t _service,
552 instance_t _instance) {
553
554 VSOMEIP_INFO << "RELEASE("
555 << std::hex << std::setw(4) << std::setfill('0') << _client << "): ["
556 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
557 << std::hex << std::setw(4) << std::setfill('0') << _instance << "]";
558
559 if (host_->get_client() == _client) {
560 std::lock_guard<std::mutex> its_lock(pending_subscription_mutex_);
561 remove_pending_subscription(_service, _instance, 0xFFFF, ANY_EVENT);
562 }
563 routing_manager_base::release_service(_client, _service, _instance);
564 requested_service_remove(_client, _service, _instance);
565
566 std::shared_ptr<serviceinfo> its_info(find_service(_service, _instance));
567 if (its_info && !its_info->is_local()) {
568 if (!its_info->get_requesters_size()) {
569 if (discovery_) {
570 discovery_->release_service(_service, _instance);
571 discovery_->unsubscribe_all(_service, _instance);
572 }
573 ep_mgr_impl_->clear_client_endpoints(_service, _instance, true);
574 ep_mgr_impl_->clear_client_endpoints(_service, _instance, false);
575 its_info->set_endpoint(nullptr, true);
576 its_info->set_endpoint(nullptr, false);
577 unset_all_eventpayloads(_service, _instance);
578 }
579 } else {
580 if (discovery_) {
581 discovery_->release_service(_service, _instance);
582 }
583 }
584}

586void routing_manager_impl::subscribe(client_t _client, uid_t _uid, gid_t _gid,
587 service_t _service, instance_t _instance, eventgroup_t _eventgroup,
588 major_version_t _major, event_t _event) {
589
590 VSOMEIP_INFO << "SUBSCRIBE("
591 << std::hex << std::setw(4) << std::setfill('0') << _client <<"): ["
592 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
593 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
594 << std::hex << std::setw(4) << std::setfill('0') << _eventgroup << ":"
595 << std::hex << std::setw(4) << std::setfill('0') << _event << ":"
596 << std::dec << (uint16_t)_major << "]";
597 const client_t its_local_client = find_local_client(_service, _instance);
598 if (get_client() == its_local_client) {
599#ifdef VSOMEIP_ENABLE_COMPAT
600 routing_manager_base::set_incoming_subscription_state(_client, _service, _instance,
601 _eventgroup, _event, subscription_state_e::IS_SUBSCRIBING);
602#endif
603 auto self = shared_from_this();
604 host_->on_subscription(_service, _instance, _eventgroup, _client, _uid, _gid, true,
605 [this, self, _client, _uid, _gid, _service, _instance, _eventgroup,
606 _event, _major]
607 (const bool _subscription_accepted) {
608 (void) ep_mgr_->find_or_create_local(_client);
609 if (!_subscription_accepted) {
610 stub_->send_subscribe_nack(_client, _service, _instance, _eventgroup, _event);
611 VSOMEIP_INFO << "Subscription request from client: 0x" << std::hex
612 << _client << std::dec << " for eventgroup: 0x" << _eventgroup
613 << " rejected from application handler.";
614 return;
615 } else {
616 stub_->send_subscribe_ack(_client, _service, _instance, _eventgroup, _event);
617 }
618 routing_manager_base::subscribe(_client, _uid, _gid, _service, _instance, _eventgroup, _major, _event);
619#ifdef VSOMEIP_ENABLE_COMPAT
620 send_pending_notify_ones(_service, _instance, _eventgroup, _client);
621 routing_manager_base::erase_incoming_subscription_state(_client, _service, _instance,
622 _eventgroup, _event);
623#endif
624 });
625 } else {
626 if (discovery_) {
627 std::set<event_t> its_already_subscribed_events;
628
629 // Note: The calls to insert_subscription & handle_subscription_state must not
630 // run concurrently to a call to on_subscribe_ack. Therefore the lock is acquired
631 // before calling insert_subscription and released after the call to
632 // handle_subscription_state.
633 std::unique_lock<std::mutex> its_critical(remote_subscription_state_mutex_);
634 bool inserted = insert_subscription(_service, _instance, _eventgroup,
635 _event, _client, &its_already_subscribed_events);
636 if (inserted) {
637 if (0 == its_local_client) {
638 handle_subscription_state(_client, _service, _instance, _eventgroup, _event);
639 its_critical.unlock();
640 static const ttl_t configured_ttl(configuration_->get_sd_ttl());
641 notify_one_current_value(_client, _service, _instance,
642 _eventgroup, _event, its_already_subscribed_events);
643
644 auto its_info = find_eventgroup(_service, _instance, _eventgroup);
645 if (its_info) {
646 discovery_->subscribe(_service, _instance, _eventgroup,
647 _major, configured_ttl,
648 its_info->is_selective() ? _client : VSOMEIP_ROUTING_CLIENT,
649 its_info);
650 }
651 } else {
652 its_critical.unlock();
653 if (is_available(_service, _instance, _major)) {
654 stub_->send_subscribe(ep_mgr_->find_local(_service, _instance),
655 _client, _service, _instance, _eventgroup, _major, _event,
656 PENDING_SUBSCRIPTION_ID);
657 }
658 }
659 }
660 if (get_client() == _client) {
661 std::lock_guard<std::mutex> ist_lock(pending_subscription_mutex_);
662 subscription_data_t subscription = {
663 _service, _instance, _eventgroup, _major, _event, _uid, _gid
664 };
665 pending_subscriptions_.insert(subscription);
666 }
667 } else {
668 VSOMEIP_ERROR<< "SOME/IP eventgroups require SD to be enabled!";
669 }
670 }

673void routing_manager_impl::unsubscribe(client_t _client, uid_t _uid, gid_t _gid,
674 service_t _service, instance_t _instance, eventgroup_t _eventgroup, event_t _event) {
675
676 VSOMEIP_INFO << "UNSUBSCRIBE("
677 << std::hex << std::setw(4) << std::setfill('0') << _client << "): ["
678 << std::hex << std::setw(4) << std::setfill('0') << _service << "."
679 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
680 << std::hex << std::setw(4) << std::setfill('0') << _eventgroup << "."
681 << std::hex << std::setw(4) << std::setfill('0') << _event << "]";
682
683 bool last_subscriber_removed(true);
684
685 std::shared_ptr<eventgroupinfo> its_info
686 = find_eventgroup(_service, _instance, _eventgroup);
687 if (its_info) {
688 for (const auto& e : its_info->get_events()) {
689 if (e->get_event() == _event || ANY_EVENT == _event)
690 e->remove_subscriber(_eventgroup, _client);
691 }
692 for (const auto& e : its_info->get_events()) {
693 if (e->has_subscriber(_eventgroup, ANY_CLIENT)) {
694 last_subscriber_removed = false;
695 break;
696 }
697 }
698 }
699
700 if (discovery_) {
701 host_->on_subscription(_service, _instance, _eventgroup, _client, _uid, _gid, false,
702 [](const bool _subscription_accepted){ (void)_subscription_accepted; });
703 if (0 == find_local_client(_service, _instance)) {
704 if (last_subscriber_removed) {
705 unset_all_eventpayloads(_service, _instance, _eventgroup);
706 {
707 auto tuple = std::make_tuple(_service, _instance, _eventgroup, _client);
708 std::lock_guard<std::mutex> its_lock(remote_subscription_state_mutex_);
709 remote_subscription_state_.erase(tuple);
710 }
711 }
712
713 if (its_info &&
714 (last_subscriber_removed || its_info->is_selective())) {
715
716 discovery_->unsubscribe(_service, _instance, _eventgroup,
717 its_info->is_selective() ? _client : VSOMEIP_ROUTING_CLIENT);
718 }
719 } else {
720 if (get_client() == _client) {
721 std::lock_guard<std::mutex> ist_lock(pending_subscription_mutex_);
722 remove_pending_subscription(_service, _instance, _eventgroup, _event);
723 stub_->send_unsubscribe(
724 ep_mgr_->find_local(_service, _instance),
725 _client, _service, _instance, _eventgroup, _event,
726 PENDING_SUBSCRIPTION_ID);
727 }
728 }
729 ep_mgr_impl_->clear_multicast_endpoints(_service, _instance);
730
731 } else {
732 VSOMEIP_ERROR<< "SOME/IP eventgroups require SD to be enabled!";
733 }
734}

740
741bool routing_manager_impl::send(client_t _client, const byte_t *_data,
742 length_t _size, instance_t _instance, bool _reliable,
743 client_t _bound_client,
744 credentials_t _credentials,
745 uint8_t _status_check, bool _sent_from_remote) {
746 bool is_sent(false);
747 if (_size > VSOMEIP_MESSAGE_TYPE_POS) {
748 std::shared_ptr<endpoint> its_target;
749 bool is_request = utility::is_request(_data[VSOMEIP_MESSAGE_TYPE_POS]);
750 bool is_notification = utility::is_notification(_data[VSOMEIP_MESSAGE_TYPE_POS]);
751 bool is_response = utility::is_response(_data[VSOMEIP_MESSAGE_TYPE_POS]);
752
753 client_t its_client = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_CLIENT_POS_MIN],
754 _data[VSOMEIP_CLIENT_POS_MAX]);
755 service_t its_service = VSOMEIP_BYTES_TO_WORD(
756 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
757 method_t its_method = VSOMEIP_BYTES_TO_WORD(
758 _data[VSOMEIP_METHOD_POS_MIN], _data[VSOMEIP_METHOD_POS_MAX]);
759
760 bool is_service_discovery
761 = (its_service == sd::service && its_method == sd::method);
762
763 if (is_request) {
764 its_target = ep_mgr_->find_local(its_service, _instance);
765 } else if (!is_notification) {
766 its_target = find_local(its_client);
767 } else if (is_notification && _client && !is_service_discovery) { // Selective notifications!
768 if (_client == get_client()) {
769#ifdef USE_DLT
770 const uint16_t its_data_size
771 = uint16_t(_size > USHRT_MAX ? USHRT_MAX : _size);
772
773 trace::header its_header;
774 if (its_header.prepare(its_target, true, _instance))
775 tc_->trace(its_header.data_, VSOMEIP_TRACE_HEADER_SIZE,
776 _data, its_data_size);
777#endif
778 deliver_message(_data, _size, _instance, _reliable, _bound_client, _credentials, _status_check, _sent_from_remote);
779 return true;
780 }
781 its_target = find_local(_client);
782 }
783
784 if (its_target) {
785#ifdef USE_DLT
786 if ((is_request && its_client == get_client()) ||
787 (is_response && find_local_client(its_service, _instance) == get_client())) {
788 const uint16_t its_data_size
789 = uint16_t(_size > USHRT_MAX ? USHRT_MAX : _size);
790
791 trace::header its_header;
792 if (its_header.prepare(its_target, true, _instance))
793 tc_->trace(its_header.data_, VSOMEIP_TRACE_HEADER_SIZE,
794 _data, its_data_size);
795 }
796#endif
797 is_sent = send_local(its_target, get_client(), _data, _size, _instance, _reliable, VSOMEIP_SEND, _status_check);
798 } else {
799 // Check whether hosting application should get the message
800 // If not, check routes to external
801 if ((its_client == host_->get_client() && is_response)
802 || (find_local_client(its_service, _instance)
803 == host_->get_client() && is_request)) {
804 // TODO: Find out how to handle session id here
805 is_sent = deliver_message(_data, _size, _instance, _reliable, VSOMEIP_ROUTING_CLIENT, _credentials, _status_check);
806 } else {
807 e2e_buffer its_buffer;
808
809 if (e2e_provider_) {
810 if ( !is_service_discovery) {
811 service_t its_service = VSOMEIP_BYTES_TO_WORD(
812 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
813 method_t its_method = VSOMEIP_BYTES_TO_WORD(
814 _data[VSOMEIP_METHOD_POS_MIN], _data[VSOMEIP_METHOD_POS_MAX]);
815#ifndef ANDROID
816 if (e2e_provider_->is_protected({its_service, its_method})) {
817 // Find out where the protected area starts
818 size_t its_base = e2e_provider_->get_protection_base({its_service, its_method});
819
820 // Build a corresponding buffer
821 its_buffer.assign(_data + its_base, _data + _size);
822
823 e2e_provider_->protect({ its_service, its_method }, its_buffer, _instance);
824
825 // Prepend header
826 its_buffer.insert(its_buffer.begin(), _data, _data + its_base);
827
828 _data = its_buffer.data();
829 }
830#endif
831 }
832 }
833 if (is_request) {
834 its_target = ep_mgr_impl_->find_or_create_remote_client(
835 its_service, _instance, _reliable);
836 if (its_target) {
837#ifdef USE_DLT
838 const uint16_t its_data_size
839 = uint16_t(_size > USHRT_MAX ? USHRT_MAX : _size);
840
841 trace::header its_header;
842 if (its_header.prepare(its_target, true, _instance))
843 tc_->trace(its_header.data_, VSOMEIP_TRACE_HEADER_SIZE,
844 _data, its_data_size);
845#endif
846 is_sent = its_target->send(_data, _size);
847 } else {
848 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
849 _data[VSOMEIP_SESSION_POS_MIN],
850 _data[VSOMEIP_SESSION_POS_MAX]);
851 VSOMEIP_ERROR<< "Routing info for remote service could not be found! ("
852 << std::hex << std::setw(4) << std::setfill('0') << its_client << "): ["
853 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
854 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
855 << std::hex << std::setw(4) << std::setfill('0') << its_method << "] "
856 << std::hex << std::setw(4) << std::setfill('0') << its_session;
857 }
858 } else {
859 std::shared_ptr<serviceinfo> its_info(find_service(its_service, _instance));
860 if (its_info || is_service_discovery) {
861 if (is_notification && !is_service_discovery) {
862 send_local_notification(get_client(), _data, _size, _instance, _reliable, _status_check);
863 method_t its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
864 _data[VSOMEIP_METHOD_POS_MAX]);
865 std::shared_ptr<event> its_event = find_event(its_service, _instance, its_method);
866 if (its_event) {
867#ifdef USE_DLT
868 bool has_sent(false);
869#endif
870 std::set<std::shared_ptr<endpoint_definition>> its_targets;
871 // we need both endpoints as clients can subscribe to events via TCP and UDP
872 std::shared_ptr<endpoint> its_udp_server_endpoint = its_info->get_endpoint(false);
873 std::shared_ptr<endpoint> its_tcp_server_endpoint = its_info->get_endpoint(true);
874
875 if (its_udp_server_endpoint || its_tcp_server_endpoint) {
876 const auto its_reliability = its_event->get_reliability();
877 for (auto its_group : its_event->get_eventgroups()) {
878 auto its_eventgroup = find_eventgroup(its_service, _instance, its_group);
879 if (its_eventgroup) {
880 // Unicast targets
881 for (const auto &its_remote : its_eventgroup->get_unicast_targets()) {
882 if (its_remote->is_reliable() && its_tcp_server_endpoint) {
883 if (its_reliability == reliability_type_e::RT_RELIABLE
884 || its_reliability == reliability_type_e::RT_BOTH) {
885 its_targets.insert(its_remote);
886 }
887 } else if (its_udp_server_endpoint && !its_eventgroup->is_sending_multicast()) {
888 if (its_reliability == reliability_type_e::RT_UNRELIABLE
889 || its_reliability == reliability_type_e::RT_BOTH) {
890 its_targets.insert(its_remote);
891 }
892 }
893 }
894 // Send to multicast targets if subscribers are still interested
895 if (its_eventgroup->is_sending_multicast()) {
896 if (its_reliability == reliability_type_e::RT_UNRELIABLE
897 || its_reliability == reliability_type_e::RT_BOTH) {
898 boost::asio::ip::address its_address;
899 uint16_t its_port;
900 if (its_eventgroup->get_multicast(its_address, its_port)) {
901 std::shared_ptr<endpoint_definition> its_multicast_target;
902 its_multicast_target = endpoint_definition::get(its_address,
903 its_port, false, its_service, _instance);
904 its_targets.insert(its_multicast_target);
905 }
906 }
907 }
908 }
909 }
910 }
911
912 for (auto const &target : its_targets) {
913 if (target->is_reliable()) {
914 its_tcp_server_endpoint->send_to(target, _data, _size);
915 } else {
916 its_udp_server_endpoint->send_to(target, _data, _size);
917 }
918#ifdef USE_DLT
919 has_sent = true;
920#endif
921 }
922#ifdef USE_DLT
923 if (has_sent) {
924 const uint16_t its_data_size
925 = uint16_t(_size > USHRT_MAX ? USHRT_MAX : _size);
926
927 trace::header its_header;
928 if (its_header.prepare(nullptr, true, _instance))
929 tc_->trace(its_header.data_, VSOMEIP_TRACE_HEADER_SIZE,
930 _data, its_data_size);
931 }
932#endif
933 }
934 } else {
935 if ((utility::is_response(_data[VSOMEIP_MESSAGE_TYPE_POS])
936 || utility::is_error(_data[VSOMEIP_MESSAGE_TYPE_POS]))
937 && !its_info->is_local()) {
938 // We received a response/error but neither the hosting application
939 // nor another local client could be found --> drop
940 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
941 _data[VSOMEIP_SESSION_POS_MIN],
942 _data[VSOMEIP_SESSION_POS_MAX]);
943 VSOMEIP_ERROR
944 << "routing_manager_impl::send: Received response/error for unknown client ("
945 << std::hex << std::setw(4) << std::setfill('0') << its_client << "): ["
946 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
947 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
948 << std::hex << std::setw(4) << std::setfill('0') << its_method << "] "
949 << std::hex << std::setw(4) << std::setfill('0') << its_session;
950 return false;
951 }
952 its_target = is_service_discovery ?
953 (sd_info_ ? sd_info_->get_endpoint(false) : nullptr) : its_info->get_endpoint(_reliable);
954 if (its_target) {
955#ifdef USE_DLT
956 const uint16_t its_data_size
957 = uint16_t(_size > USHRT_MAX ? USHRT_MAX : _size);
958
959 trace::header its_header;
960 if (its_header.prepare(its_target, true, _instance))
961 tc_->trace(its_header.data_, VSOMEIP_TRACE_HEADER_SIZE,
962 _data, its_data_size);
963#endif
964 is_sent = its_target->send(_data, _size);
965 } else {
966 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
967 _data[VSOMEIP_SESSION_POS_MIN],
968 _data[VSOMEIP_SESSION_POS_MAX]);
969 VSOMEIP_ERROR << "Routing error. Endpoint for service ("
970 << std::hex << std::setw(4) << std::setfill('0') << its_client << "): ["
971 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
972 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
973 << std::hex << std::setw(4) << std::setfill('0') << its_method << "] "
974 << std::hex << std::setw(4) << std::setfill('0') << its_session
975 << " could not be found!";
976 }
977 }
978 } else {
979 if (!is_notification) {
980 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
981 _data[VSOMEIP_SESSION_POS_MIN],
982 _data[VSOMEIP_SESSION_POS_MAX]);
983 VSOMEIP_ERROR << "Routing error. Not hosting service ("
984 << std::hex << std::setw(4) << std::setfill('0') << its_client << "): ["
985 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
986 << std::hex << std::setw(4) << std::setfill('0') << _instance << "."
987 << std::hex << std::setw(4) << std::setfill('0') << its_method << "] "
988 << std::hex << std::setw(4) << std::setfill('0') << its_session;
989 }
990 }
991 }
992 }
993 }
994 }
995
996 return (is_sent);
997}

1337 _data[VSOMEIP_SERVICE_POS_MAX]);

1451#endif

1622 ready_to_stop_t() : reliable_(false), unreliable_(false){}
1623 std::atomic<bool> reliable_;
1624 std::atomic<bool> unreliable_;
1625 };
1626 auto ready_to_stop = std::make_shared<ready_to_stop_t>();
1627 auto ptr = shared_from_this();

1693 }
1694 };
1695
1696 // determine callback count
1697 for (const auto& ep : {its_reliable_endpoint, its_unreliable_endpoint}) {
1698 if (ep) {
1699 std::lock_guard<std::mutex> its_lock(callback_counts_mutex_);
1700 auto found_id = callback_counts_.find(its_callback_id);
1701 if (found_id != callback_counts_.end()) {
1702 found_id->second++;
1703 } else {
1704 callback_counts_[its_callback_id] = 1;
1705 }
1706 }
1707 }
1708 for (const auto& ep : {its_reliable_endpoint, its_unreliable_endpoint}) {
1709 if (ep) {
1710 ep->prepare_stop(callback, _service);
1711 }
1712 }
1713
1714 if (!its_reliable_endpoint && !its_unreliable_endpoint) {
1715 {
1716 std::lock_guard<std::mutex> its_lock(callback_counts_mutex_);
1717 callback_counts_.erase(its_callback_id);
1718 }
1719 erase_offer_command(_service, _instance);
1720 }
1721
1722 std::lock_guard<std::mutex> its_eventgroups_lock(eventgroups_mutex_);
1723 auto find_service = eventgroups_.find(_service);
1724 if (find_service != eventgroups_.end()) {
1725 auto find_instance = find_service->second.find(_instance);
1726 if (find_instance != find_service->second.end()) {
1727 for (auto e : find_instance->second) {
1728 e.second->clear_remote_subscriptions();
1729 }
1730 }
1731 }
1732 } else {
1733 erase_offer_command(_service, _instance);
1734 }
1735}

2224 {

2240 // service is marked as available later in on_connect()

2242 if (udp_inserted) {
2243 // atomically create reliable and unreliable endpoint
2244 ep_mgr_impl_->find_or_create_remote_client(
