25tcp_server_endpoint_impl::tcp_server_endpoint_impl(
26 const std::shared_ptr<endpoint_host>& _endpoint_host,
27 const std::shared_ptr<routing_host>& _routing_host,
28 const endpoint_type& _local,
29 boost::asio::io_service &_io,
30 const std::shared_ptr<configuration>& _configuration)
31 : tcp_server_endpoint_base_impl(_endpoint_host, _routing_host, _local, _io,
32 _configuration->get_max_message_size_reliable(_local.address().to_string(), _local.port()),
33 _configuration->get_endpoint_queue_limit(_local.address().to_string(), _local.port()),
34 _configuration),
35 acceptor_(_io),
36 buffer_shrink_threshold_(configuration_->get_buffer_shrink_threshold()),
37 local_port_(_local.port()),
38 // send timeout after 2/3 of configured ttl, warning after 1/3
39 send_timeout_(configuration_->get_sd_ttl() * 666) {
40 is_supporting_magic_cookies_ = true;
41
42 boost::system::error_code ec;
43 acceptor_.open(_local.protocol(), ec);
44 boost::asio::detail::throw_error(ec, "acceptor open");
45 acceptor_.set_option(boost::asio::socket_base::reuse_address(true), ec);
46 boost::asio::detail::throw_error(ec, "acceptor set_option");
47
48#ifndef _WIN32
49 // If specified, bind to device
50 std::string its_device(configuration_->get_device());
51 if (its_device != "") {
52 if (setsockopt(acceptor_.native_handle(),
53 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), (int)its_device.size()) == -1) {
54 VSOMEIP_WARNING << "TCP Server: Could not bind to device \"" << its_device << "\"";
55 }
56 }
57#endif
58
59 acceptor_.bind(_local, ec);
60 boost::asio::detail::throw_error(ec, "acceptor bind");
61 acceptor_.listen(boost::asio::socket_base::max_connections, ec);
62 boost::asio::detail::throw_error(ec, "acceptor listen");
63}

72void tcp_server_endpoint_impl::start() {
73 std::lock_guard<std::mutex> its_lock(acceptor_mutex_);
74 if (acceptor_.is_open()) {
75 connection::ptr new_connection = connection::create(
76 std::dynamic_pointer_cast<tcp_server_endpoint_impl>(
77 shared_from_this()), max_message_size_,
78 buffer_shrink_threshold_, has_enabled_magic_cookies_,
79 service_, send_timeout_);
80
81 {
82 std::unique_lock<std::mutex> its_socket_lock(new_connection->get_socket_lock());
83 acceptor_.async_accept(new_connection->get_socket(),
84 std::bind(&tcp_server_endpoint_impl::accept_cbk,
85 std::dynamic_pointer_cast<tcp_server_endpoint_impl>(
86 shared_from_this()), new_connection,
87 std::placeholders::_1));
88 }
89 }

118bool tcp_server_endpoint_impl::send_error(
119 const std::shared_ptr<endpoint_definition> _target,
120 const byte_t *_data, uint32_t _size) {
121 bool ret(false);
122 std::lock_guard<std::mutex> its_lock(mutex_);
123 const endpoint_type its_target(_target->get_address(), _target->get_port());
124 const queue_iterator_type target_queue_iterator(find_or_create_queue_unlocked(its_target));
125 auto& its_qpair = target_queue_iterator->second;
126 const bool queue_size_zero_on_entry(its_qpair.second.empty());
127
128 if (check_message_size(nullptr, _size, its_target) == endpoint_impl::cms_ret_e::MSG_OK &&
129 check_queue_limit(_data, _size, its_qpair.first)) {
130 its_qpair.second.emplace_back(
131 std::make_shared<message_buffer_t>(_data, _data + _size));
132 its_qpair.first += _size;
133
134 if (queue_size_zero_on_entry) { // no writing in progress
135 send_queued(target_queue_iterator);
136 }
137 ret = true;
138 }
139 return ret;
140}

142void tcp_server_endpoint_impl::send_queued(const queue_iterator_type _queue_iterator) {
143 connection::ptr its_connection;
144 {
145 std::lock_guard<std::mutex> its_lock(connections_mutex_);
146 auto connection_iterator = connections_.find(_queue_iterator->first);
147 if (connection_iterator != connections_.end()) {
148 its_connection = connection_iterator->second;
149 } else {
150 VSOMEIP_INFO << "Didn't find connection: "
151 << _queue_iterator->first.address().to_string() << ":" << std::dec
152 << static_cast<std::uint16_t>(_queue_iterator->first.port())
153 << " dropping outstanding messages (" << std::dec
154 << _queue_iterator->second.second.size() << ").";
155 queues_.erase(_queue_iterator->first);
156 }
157 }
158 if (its_connection) {
159 its_connection->send_queued(_queue_iterator);
160 }

208void tcp_server_endpoint_impl::accept_cbk(const connection::ptr& _connection,
209 boost::system::error_code const &_error) {
210
211 if (!_error) {
212 boost::system::error_code its_error;
213 endpoint_type remote;
214 {
215 std::unique_lock<std::mutex> its_socket_lock(_connection->get_socket_lock());
216 socket_type &new_connection_socket = _connection->get_socket();
217 remote = new_connection_socket.remote_endpoint(its_error);
218 _connection->set_remote_info(remote);
219 // Nagle algorithm off
220 new_connection_socket.set_option(ip::tcp::no_delay(true), its_error);
221
222 new_connection_socket.set_option(boost::asio::socket_base::keep_alive(true), its_error);
223 if (its_error) {
224 VSOMEIP_WARNING << "tcp_server_endpoint::connect: couldn't enable "
225 << "keep_alive: " << its_error.message();
226 }
227 }
228 if (!its_error) {
229 {
230 std::lock_guard<std::mutex> its_lock(connections_mutex_);
231 connections_[remote] = _connection;
232 }
233 _connection->start();
234 }
235 }
236 if (_error != boost::asio::error::bad_descriptor
237 && _error != boost::asio::error::operation_aborted
238 && _error != boost::asio::error::no_descriptors) {
239 start();
240 } else if (_error == boost::asio::error::no_descriptors) {
241 VSOMEIP_ERROR<< "tcp_server_endpoint_impl::accept_cbk: "
242 << _error.message() << " (" << std::dec << _error.value()
243 << ") Will try to accept again in 1000ms";
244 std::shared_ptr<boost::asio::steady_timer> its_timer =
245 std::make_shared<boost::asio::steady_timer>(service_,
246 std::chrono::milliseconds(1000));
247 auto its_ep = std::dynamic_pointer_cast<tcp_server_endpoint_impl>(
248 shared_from_this());
249 its_timer->async_wait([its_timer, its_ep]
250 (const boost::system::error_code& _error) {
251 if (!_error) {
252 its_ep->start();
253 }
254 });
255 }
256}

323void tcp_server_endpoint_impl::connection::receive() {
324 std::lock_guard<std::mutex> its_lock(socket_mutex_);
325 if(socket_.is_open()) {
326 const std::size_t its_capacity(recv_buffer_.capacity());
327 size_t buffer_size = its_capacity - recv_buffer_size_;
328 try {
329 if (missing_capacity_) {
330 if (missing_capacity_ > MESSAGE_SIZE_UNLIMITED) {
331 VSOMEIP_ERROR << "Missing receive buffer capacity exceeds allowed maximum!";
332 return;
333 }
334 const std::size_t its_required_capacity(recv_buffer_size_ + missing_capacity_);
335 if (its_capacity < its_required_capacity) {
336 recv_buffer_.reserve(its_required_capacity);
337 recv_buffer_.resize(its_required_capacity, 0x0);
338 if (recv_buffer_.size() > 1048576) {
339 VSOMEIP_INFO << "tse: recv_buffer size is: " <<
340 recv_buffer_.size()
341 << " local: " << get_address_port_local()
342 << " remote: " << get_address_port_remote();
343 }
344 }
345 buffer_size = missing_capacity_;
346 missing_capacity_ = 0;
347 } else if (buffer_shrink_threshold_
348 && shrink_count_ > buffer_shrink_threshold_
349 && recv_buffer_size_ == 0) {
350 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
351 recv_buffer_.shrink_to_fit();
352 buffer_size = recv_buffer_size_initial_;
353 shrink_count_ = 0;
354 }
355 } catch (const std::exception &e) {
356 handle_recv_buffer_exception(e);
357 // don't start receiving again
358 return;
359 }
360 socket_.async_receive(boost::asio::buffer(&recv_buffer_[recv_buffer_size_], buffer_size),
361 std::bind(&tcp_server_endpoint_impl::connection::receive_cbk,
362 shared_from_this(), std::placeholders::_1,
363 std::placeholders::_2));
364 }
365}

376void tcp_server_endpoint_impl::connection::send_queued(
377 const queue_iterator_type _queue_iterator) {
378 std::shared_ptr<tcp_server_endpoint_impl> its_server(server_.lock());
379 if (!its_server) {
380 VSOMEIP_TRACE << "tcp_server_endpoint_impl::connection::send_queued "
381 " couldn't lock server_";
382 return;
383 }
384 message_buffer_ptr_t its_buffer = _queue_iterator->second.second.front();
385 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
386 (*its_buffer)[VSOMEIP_SERVICE_POS_MIN],
387 (*its_buffer)[VSOMEIP_SERVICE_POS_MAX]);
388 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
389 (*its_buffer)[VSOMEIP_METHOD_POS_MIN],
390 (*its_buffer)[VSOMEIP_METHOD_POS_MAX]);
391 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
392 (*its_buffer)[VSOMEIP_CLIENT_POS_MIN],
393 (*its_buffer)[VSOMEIP_CLIENT_POS_MAX]);
394 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
395 (*its_buffer)[VSOMEIP_SESSION_POS_MIN],
396 (*its_buffer)[VSOMEIP_SESSION_POS_MAX]);
397 if (magic_cookies_enabled_) {
398 const std::chrono::steady_clock::time_point now =
399 std::chrono::steady_clock::now();
400 if (std::chrono::duration_cast<std::chrono::milliseconds>(
401 now - last_cookie_sent_) > std::chrono::milliseconds(10000)) {
402 if (send_magic_cookie(its_buffer)) {
403 last_cookie_sent_ = now;
404 _queue_iterator->second.first += sizeof(SERVICE_COOKIE);
405 }
406 }
407 }
408
409 {
410 std::lock_guard<std::mutex> its_lock(socket_mutex_);
411 {
412 std::lock_guard<std::mutex> its_sent_lock(its_server->sent_mutex_);
413 its_server->is_sending_ = true;
414 }
415
416 boost::asio::async_write(socket_, boost::asio::buffer(*its_buffer),
417 std::bind(&tcp_server_endpoint_impl::connection::write_completion_condition,
418 shared_from_this(),
419 std::placeholders::_1,
420 std::placeholders::_2,
421 its_buffer->size(),
422 its_service, its_method, its_client, its_session,
423 std::chrono::steady_clock::now()),
424 std::bind(&tcp_server_endpoint_base_impl::send_cbk,
425 its_server,
426 _queue_iterator,
427 std::placeholders::_1,
428 std::placeholders::_2));
429 }
430}

432void tcp_server_endpoint_impl::connection::send_queued_sync(
433 const queue_iterator_type _queue_iterator) {
434 message_buffer_ptr_t its_buffer = _queue_iterator->second.second.front();
435 if (magic_cookies_enabled_) {
436 const std::chrono::steady_clock::time_point now =
437 std::chrono::steady_clock::now();
438 if (std::chrono::duration_cast<std::chrono::milliseconds>(
439 now - last_cookie_sent_) > std::chrono::milliseconds(10000)) {
440 if (send_magic_cookie(its_buffer)) {
441 last_cookie_sent_ = now;
442 _queue_iterator->second.first += sizeof(SERVICE_COOKIE);
443 }
444 }
445 }
446
447 try {
448 std::lock_guard<std::mutex> its_lock(socket_mutex_);
449 boost::asio::write(socket_, boost::asio::buffer(*its_buffer));
450 } catch (const boost::system::system_error &e) {
451 if (e.code() != boost::asio::error::broken_pipe) {
452 VSOMEIP_ERROR << "tcp_server_endpoint_impl::connection::"
453 << __func__ << " " << e.what();
454 }
455 }

458bool tcp_server_endpoint_impl::connection::send_magic_cookie(
459 message_buffer_ptr_t &_buffer) {
460 if (max_message_size_ == MESSAGE_SIZE_UNLIMITED
461 || max_message_size_ - _buffer->size() >=
462 VSOMEIP_SOMEIP_HEADER_SIZE + VSOMEIP_SOMEIP_MAGIC_COOKIE_SIZE) {
463 _buffer->insert(_buffer->begin(), SERVICE_COOKIE,
464 SERVICE_COOKIE + sizeof(SERVICE_COOKIE));
465 return true;
466 }
467 return false;
468}

475void tcp_server_endpoint_impl::connection::receive_cbk(
476 boost::system::error_code const &_error,
477 std::size_t _bytes) {
478 if (_error == boost::asio::error::operation_aborted) {
479 // endpoint was stopped
480 return;
481 }
482 std::shared_ptr<tcp_server_endpoint_impl> its_server(server_.lock());
483 if (!its_server) {
484 VSOMEIP_ERROR << "tcp_server_endpoint_impl::connection::receive_cbk "
485 " couldn't lock server_";
486 return;
487 }
488#if 0
489 std::stringstream msg;
490 for (std::size_t i = 0; i < _bytes + recv_buffer_size_; ++i)
491 msg << std::hex << std::setw(2) << std::setfill('0')
492 << (int) recv_buffer_[i] << " ";
493 VSOMEIP_INFO << msg.str();
494#endif
495 std::shared_ptr<routing_host> its_host = its_server->routing_host_.lock();
496 if (its_host) {
497 if (!_error && 0 < _bytes) {
498 if (recv_buffer_size_ + _bytes < recv_buffer_size_) {
499 VSOMEIP_ERROR << "receive buffer overflow in tcp client endpoint ~> abort!";
500 return;
501 }
502 recv_buffer_size_ += _bytes;
503
504 size_t its_iteration_gap = 0;
505 bool has_full_message;
506 do {
507 uint64_t read_message_size
508 = utility::get_message_size(&recv_buffer_[its_iteration_gap],
509 recv_buffer_size_);
510 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
511 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
512 return;
513 }
514 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
515 has_full_message = (current_message_size > VSOMEIP_RETURN_CODE_POS
516 && current_message_size <= recv_buffer_size_);
517 if (has_full_message) {
518 bool needs_forwarding(true);
519 if (is_magic_cookie(its_iteration_gap)) {
520 magic_cookies_enabled_ = true;
521 } else {
522 if (magic_cookies_enabled_) {
523 uint32_t its_offset
524 = its_server->find_magic_cookie(&recv_buffer_[its_iteration_gap],
525 recv_buffer_size_);
526 if (its_offset < current_message_size) {
527 {
528 std::lock_guard<std::mutex> its_lock(socket_mutex_);
529 VSOMEIP_ERROR << "Detected Magic Cookie within message data. Resyncing."
530 << " local: " << get_address_port_local()
531 << " remote: " << get_address_port_remote();
532 }
533 if (!is_magic_cookie(its_iteration_gap)) {
534 auto its_endpoint_host = its_server->endpoint_host_.lock();
535 if (its_endpoint_host) {
536 its_endpoint_host->on_error(&recv_buffer_[its_iteration_gap],
537 static_cast<length_t>(recv_buffer_size_),its_server.get(),
538 remote_address_, remote_port_);
539 }
540 }
541 current_message_size = its_offset;
542 needs_forwarding = false;
543 }
544 }
545 }
546 if (needs_forwarding) {
547 if (utility::is_request(
548 recv_buffer_[its_iteration_gap
549 + VSOMEIP_MESSAGE_TYPE_POS])) {
550 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
551 recv_buffer_[its_iteration_gap + VSOMEIP_CLIENT_POS_MIN],
552 recv_buffer_[its_iteration_gap + VSOMEIP_CLIENT_POS_MAX]);
553 if (its_client != MAGIC_COOKIE_CLIENT) {
554 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
555 recv_buffer_[its_iteration_gap + VSOMEIP_SESSION_POS_MIN],
556 recv_buffer_[its_iteration_gap + VSOMEIP_SESSION_POS_MAX]);
557 its_server->clients_mutex_.lock();
558 its_server->clients_[its_client][its_session] = remote_;
559 its_server->clients_mutex_.unlock();
560 }
561 }
562 if (!magic_cookies_enabled_) {
563 its_host->on_message(&recv_buffer_[its_iteration_gap],
564 current_message_size, its_server.get(),
565 boost::asio::ip::address(),
566 VSOMEIP_ROUTING_CLIENT,
567 std::make_pair(ANY_UID, ANY_GID),
568 remote_address_, remote_port_);
569 } else {
570 // Only call on_message without a magic cookie in front of the buffer!
571 if (!is_magic_cookie(its_iteration_gap)) {
572 its_host->on_message(&recv_buffer_[its_iteration_gap],
573 current_message_size, its_server.get(),
574 boost::asio::ip::address(),
575 VSOMEIP_ROUTING_CLIENT,
576 std::make_pair(ANY_UID, ANY_GID),
577 remote_address_, remote_port_);
578 }
579 }
580 }
581 calculate_shrink_count();
582 missing_capacity_ = 0;
583 recv_buffer_size_ -= current_message_size;
584 its_iteration_gap += current_message_size;
585 } else if (magic_cookies_enabled_ && recv_buffer_size_ > 0) {
586 uint32_t its_offset =
587 its_server->find_magic_cookie(&recv_buffer_[its_iteration_gap],
588 recv_buffer_size_);
589 if (its_offset < recv_buffer_size_) {
590 {
591 std::lock_guard<std::mutex> its_lock(socket_mutex_);
592 VSOMEIP_ERROR << "Detected Magic Cookie within message data. Resyncing."
593 << " local: " << get_address_port_local()
594 << " remote: " << get_address_port_remote();
595 }
596 if (!is_magic_cookie(its_iteration_gap)) {
597 auto its_endpoint_host = its_server->endpoint_host_.lock();
598 if (its_endpoint_host) {
599 its_endpoint_host->on_error(&recv_buffer_[its_iteration_gap],
600 static_cast<length_t>(recv_buffer_size_), its_server.get(),
601 remote_address_, remote_port_);
602 }
603 }
604 recv_buffer_size_ -= its_offset;
605 its_iteration_gap += its_offset;
606 has_full_message = true; // trigger next loop
607 if (!is_magic_cookie(its_iteration_gap)) {
608 auto its_endpoint_host = its_server->endpoint_host_.lock();
609 if (its_endpoint_host) {
610 its_endpoint_host->on_error(&recv_buffer_[its_iteration_gap],
611 static_cast<length_t>(recv_buffer_size_), its_server.get(),
612 remote_address_, remote_port_);
613 }
614 }
615 }
616 }
617
618 if (!has_full_message) {
619 if (recv_buffer_size_ > VSOMEIP_RETURN_CODE_POS &&
620 (recv_buffer_[its_iteration_gap + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION ||
621 !utility::is_valid_message_type(static_cast<message_type_e>(recv_buffer_[its_iteration_gap + VSOMEIP_MESSAGE_TYPE_POS])) ||
622 !utility::is_valid_return_code(static_cast<return_code_e>(recv_buffer_[its_iteration_gap + VSOMEIP_RETURN_CODE_POS]))
623 )) {
624 if (recv_buffer_[its_iteration_gap + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
625 {
626 std::lock_guard<std::mutex> its_lock(socket_mutex_);
627 VSOMEIP_ERROR << "tse: Wrong protocol version: 0x"
628 << std::hex << std::setw(2) << std::setfill('0')
629 << std::uint32_t(recv_buffer_[its_iteration_gap + VSOMEIP_PROTOCOL_VERSION_POS])
630 << " local: " << get_address_port_local()
631 << " remote: " << get_address_port_remote()
632 << ". Closing connection due to missing/broken data TCP stream.";
633 }
634 // ensure to send back a error message w/ wrong protocol version
635 its_host->on_message(&recv_buffer_[its_iteration_gap],
636 VSOMEIP_SOMEIP_HEADER_SIZE + 8, its_server.get(),
637 boost::asio::ip::address(),
638 VSOMEIP_ROUTING_CLIENT,
639 std::make_pair(ANY_UID, ANY_GID),
640 remote_address_, remote_port_);
641 } else if (!utility::is_valid_message_type(static_cast<message_type_e>(
642 recv_buffer_[its_iteration_gap + VSOMEIP_MESSAGE_TYPE_POS]))) {
643 std::lock_guard<std::mutex> its_lock(socket_mutex_);
644 VSOMEIP_ERROR << "tse: Invalid message type: 0x"
645 << std::hex << std::setw(2) << std::setfill('0')
646 << std::uint32_t(recv_buffer_[its_iteration_gap + VSOMEIP_MESSAGE_TYPE_POS])
647 << " local: " << get_address_port_local()
648 << " remote: " << get_address_port_remote()
649 << ". Closing connection due to missing/broken data TCP stream.";
650 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
651 recv_buffer_[its_iteration_gap + VSOMEIP_RETURN_CODE_POS]))) {
652 std::lock_guard<std::mutex> its_lock(socket_mutex_);
653 VSOMEIP_ERROR << "tse: Invalid return code: 0x"
654 << std::hex << std::setw(2) << std::setfill('0')
655 << std::uint32_t(recv_buffer_[its_iteration_gap + VSOMEIP_RETURN_CODE_POS])
656 << " local: " << get_address_port_local()
657 << " remote: " << get_address_port_remote()
658 << ". Closing connection due to missing/broken data TCP stream.";
659 }
660 wait_until_sent(boost::asio::error::operation_aborted);
661 return;
662 } else if (max_message_size_ != MESSAGE_SIZE_UNLIMITED
663 && current_message_size > max_message_size_) {
664 recv_buffer_size_ = 0;
665 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
666 recv_buffer_.shrink_to_fit();
667 if (magic_cookies_enabled_) {
668 std::lock_guard<std::mutex> its_lock(socket_mutex_);
669 VSOMEIP_ERROR << "Received a TCP message which exceeds "
670 << "maximum message size ("
671 << std::dec << current_message_size
672 << " > " << std::dec << max_message_size_
673 << "). Magic Cookies are enabled: "
674 << "Resetting receiver. local: "
675 << get_address_port_local() << " remote: "
676 << get_address_port_remote();
677 } else {
678 {
679 std::lock_guard<std::mutex> its_lock(socket_mutex_);
680 VSOMEIP_ERROR << "Received a TCP message which exceeds "
681 << "maximum message size ("
682 << std::dec << current_message_size
683 << " > " << std::dec << max_message_size_
684 << ") Magic cookies are disabled: "
685 << "Connection will be closed! local: "
686 << get_address_port_local() << " remote: "
687 << get_address_port_remote();
688 }
689 wait_until_sent(boost::asio::error::operation_aborted);
690 return;
691 }
692 } else if (current_message_size > recv_buffer_size_) {
693 missing_capacity_ = current_message_size
694 - static_cast<std::uint32_t>(recv_buffer_size_);
695 } else if (VSOMEIP_SOMEIP_HEADER_SIZE > recv_buffer_size_) {
696 missing_capacity_ = VSOMEIP_SOMEIP_HEADER_SIZE
697 - static_cast<std::uint32_t>(recv_buffer_size_);
698 } else if (magic_cookies_enabled_ && recv_buffer_size_ > 0) {
699 // no need to check for magic cookie here again: has_full_message
700 // would have been set to true if there was one present in the data
701 recv_buffer_size_ = 0;
702 recv_buffer_.resize(recv_buffer_size_initial_, 0x0);
703 recv_buffer_.shrink_to_fit();
704 missing_capacity_ = 0;
705 std::lock_guard<std::mutex> its_lock(socket_mutex_);
706 VSOMEIP_ERROR << "Didn't find magic cookie in broken"
707 << " data, trying to resync."
708 << " local: " << get_address_port_local()
709 << " remote: " << get_address_port_remote();
710 } else {
711 {
712 std::lock_guard<std::mutex> its_lock(socket_mutex_);
713 VSOMEIP_ERROR << "tse::c<" << this
714 << ">rcb: recv_buffer_size is: " << std::dec
715 << recv_buffer_size_ << " but couldn't read "
716 "out message_size. recv_buffer_capacity: "
717 << recv_buffer_.capacity()
718 << " its_iteration_gap: " << its_iteration_gap
719 << "local: " << get_address_port_local()
720 << " remote: " << get_address_port_remote()
721 << ". Closing connection due to missing/broken data TCP stream.";
722 }
723 wait_until_sent(boost::asio::error::operation_aborted);
724 return;
725 }
726 }
727 } while (has_full_message && recv_buffer_size_);
728 if (its_iteration_gap) {
729 // Copy incomplete message to front for next receive_cbk iteration
730 for (size_t i = 0; i < recv_buffer_size_; ++i) {
731 recv_buffer_[i] = recv_buffer_[i + its_iteration_gap];
732 }
733 // Still more capacity needed after shifting everything to front?
734 if (missing_capacity_ &&
735 missing_capacity_ <= recv_buffer_.capacity() - recv_buffer_size_) {
736 missing_capacity_ = 0;
737 }
738 }
739 receive();
740 }
741 }
742 if (_error == boost::asio::error::eof
743 || _error == boost::asio::error::connection_reset
744 || _error == boost::asio::error::timed_out) {
745 if(_error == boost::asio::error::timed_out) {
746 std::lock_guard<std::mutex> its_lock(socket_mutex_);
747 VSOMEIP_WARNING << "tcp_server_endpoint receive_cbk: " << _error.message()
748 << " local: " << get_address_port_local()
749 << " remote: " << get_address_port_remote();
750 }
751 wait_until_sent(boost::asio::error::operation_aborted);
752 }
753}
