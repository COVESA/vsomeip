162 // Set a socket option.
163 template <typename Option>
164 boost::system::error_code get_option(const implementation_type& impl,
165 Option& option, boost::system::error_code& ec) const
166 {
167 std::size_t size = option.size(impl.protocol_);
168 socket_ops::getsockopt(impl.socket_, impl.state_,
169 option.level(impl.protocol_), option.name(impl.protocol_),
170 option.data(impl.protocol_), &size, ec);
171 if (!ec)
172 option.resize(impl.protocol_, size);
173 return ec;

182 if (socket_ops::getsockname(impl.socket_, endpoint.data(), &addr_len, ec))
183 return endpoint_type();
184 endpoint.resize(addr_len);
185 return endpoint;

191 {
192 endpoint_type endpoint;
193 std::size_t addr_len = endpoint.capacity();
194 if (socket_ops::getpeername(impl.socket_,
195 endpoint.data(), &addr_len, false, ec))
196 return endpoint_type();
197 endpoint.resize(addr_len);
198 return endpoint;

230 void async_send_to(implementation_type& impl,
231 const ConstBufferSequence& buffers,

233 Handler& handler)
234 {
235 bool is_continuation =
236 boost_asio_handler_cont_helpers::is_continuation(handler);
237
238 // Allocate and construct an operation to wrap the handler.

240 endpoint_type, Handler> op;

242 boost_asio_handler_alloc_helpers::allocate(

249 p.v = p.p = 0;

274 // Receive a datagram with the endpoint of the sender. Returns the number of
275 // bytes received.
276 template <typename MutableBufferSequence>
277 size_t receive_from(implementation_type& impl,
278 const MutableBufferSequence& buffers,
279 endpoint_type& sender_endpoint, socket_base::message_flags flags,
280 boost::system::error_code& ec)
281 {
282 buffer_sequence_adapter<boost::asio::mutable_buffer,
283 MutableBufferSequence> bufs(buffers);
284
285 std::size_t addr_len = sender_endpoint.capacity();
286 std::size_t bytes_recvd = socket_ops::sync_recvfrom(
287 impl.socket_, impl.state_, bufs.buffers(), bufs.count(),
288 flags, sender_endpoint.data(), &addr_len, ec);
289
290 if (!ec)
291 sender_endpoint.resize(addr_len);
292
293 return bytes_recvd;
294 }
295
296 // Wait until data can be received without blocking.
297 size_t receive_from(implementation_type& impl, const null_buffers&,
298 endpoint_type& sender_endpoint, socket_base::message_flags,
299 boost::system::error_code& ec)
300 {
301 // Wait for socket to become ready.
302 socket_ops::poll_read(impl.socket_, impl.state_, ec);
303
304 // Reset endpoint since it can be given no sensible value at this time.
305 sender_endpoint = endpoint_type();
306
307 return 0;
308 }

314 void async_receive_from(implementation_type& impl,
315 const MutableBufferSequence& buffers, endpoint_type& sender_endpoint,
316 socket_base::message_flags flags, Handler& handler)
317 {

320
321 // Allocate and construct an operation to wrap the handler.
322 typedef reactive_socket_recvfrom_op_ext_local<MutableBufferSequence,
323 endpoint_type, Handler> op;
324 typename op::ptr p = { boost::asio::detail::addressof(handler),

326 sizeof(op), handler), 0 };

328 p.p = new (p.v) op(impl.socket_, protocol,

338 p.v = p.p = 0;

370 // Accept a new connection.
371 template <typename Socket>
372 boost::system::error_code accept(implementation_type& impl,
373 Socket& peer, endpoint_type* peer_endpoint, boost::system::error_code& ec)
374 {
375 // We cannot accept a socket that is already open.
376 if (peer.is_open())
377 {
378 ec = boost::asio::error::already_open;
379 return ec;
380 }
381
382 std::size_t addr_len = peer_endpoint ? peer_endpoint->capacity() : 0;
383 socket_holder new_socket(socket_ops::sync_accept(impl.socket_,
384 impl.state_, peer_endpoint ? peer_endpoint->data() : 0,
385 peer_endpoint ? &addr_len : 0, ec));
386
387 // On success, assign new connection to peer socket object.
388 if (new_socket.get() != invalid_socket)
389 {
390 if (peer_endpoint)
391 peer_endpoint->resize(addr_len);
392 if (!peer.assign(impl.protocol_, new_socket.get(), ec))
393 new_socket.release();
394 }
395
396 return ec;
397 }
