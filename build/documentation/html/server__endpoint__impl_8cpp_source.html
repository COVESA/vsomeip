46template<typename Protocol>
47void server_endpoint_impl<Protocol>::prepare_stop(
48 endpoint::prepare_stop_handler_t _handler, service_t _service) {
49 std::lock_guard<std::mutex> its_lock(mutex_);
50 bool queued_train(false);
51 if (_service == ANY_SERVICE) { // endpoint is shutting down completely
52 endpoint_impl<Protocol>::sending_blocked_ = true;
53 boost::system::error_code ec;
54 for (auto const& train_iter : trains_) {
55 train_iter.second->departure_timer_->cancel(ec);
56 if (train_iter.second->buffer_->size() > 0) {
57 auto target_queue_iter = queues_.find(train_iter.first);
58 if (target_queue_iter != queues_.end()) {
59 auto& its_qpair = target_queue_iter->second;
60 const bool queue_size_zero_on_entry(its_qpair.second.empty());
61 queue_train(target_queue_iter, train_iter.second,
62 queue_size_zero_on_entry);
63 queued_train = true;
64 }
65 }
66 }
67 } else {
68 for (auto const& train_iter : trains_) {
69 for (auto const& passenger_iter : train_iter.second->passengers_) {
70 if (passenger_iter.first == _service) {
71 // cancel departure timer
72 boost::system::error_code ec;
73 train_iter.second->departure_timer_->cancel(ec);
74 // queue train
75 auto target_queue_iter = queues_.find(train_iter.first);
76 if (target_queue_iter != queues_.end()) {
77 const auto& its_qpair = target_queue_iter->second;
78 const bool queue_size_zero_on_entry(its_qpair.second.empty());
79 queue_train(target_queue_iter, train_iter.second,
80 queue_size_zero_on_entry);
81 queued_train = true;
82 }
83 break;
84 }
85 }
86 }
87 }
88 if (!queued_train) {
89 if (_service == ANY_SERVICE) {
90 if (std::all_of(queues_.begin(), queues_.end(),
91 [&](const typename queue_type::value_type& q)
92 { return q.second.second.empty(); })) {
93 // nothing was queued and all queues are empty -> ensure cbk is called
94 auto ptr = this->shared_from_this();
95 endpoint_impl<Protocol>::service_.post([ptr, _handler, _service](){
96 _handler(ptr, _service);
97 });
98 } else {
99 prepare_stop_handlers_[_service] = _handler;
100 }
101 } else {
102 // check if any of the queues contains a message of to be stopped service
103 bool found_service_msg(false);
104 for (const auto& q : queues_) {
105 for (const auto& msg : q.second.second ) {
106 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
107 (*msg)[VSOMEIP_SERVICE_POS_MIN],
108 (*msg)[VSOMEIP_SERVICE_POS_MAX]);
109 if (its_service == _service) {
110 found_service_msg = true;
111 break;
112 }
113 }
114 if (found_service_msg) {
115 break;
116 }
117 }
118 if (found_service_msg) {
119 prepare_stop_handlers_[_service] = _handler;
120 } else { // no messages of the to be stopped service are or have been queued
121 auto ptr = this->shared_from_this();
122 endpoint_impl<Protocol>::service_.post([ptr, _handler, _service](){
123 _handler(ptr, _service);
124 });
125 }
126 }
127 } else {
128 prepare_stop_handlers_[_service] = _handler;
129 }
130}

167template<typename Protocol>bool server_endpoint_impl<Protocol>::send(const uint8_t *_data,
168 uint32_t _size) {
169#if 0
170 std::stringstream msg;
171 msg << "sei::send ";
172 for (uint32_t i = 0; i < _size; i++)
173 msg << std::setw(2) << std::setfill('0') << (int)_data[i] << " ";
174 VSOMEIP_INFO << msg.str();
175#endif
176 endpoint_type its_target;
177 bool is_valid_target(false);
178
179 if (VSOMEIP_SESSION_POS_MAX < _size) {
180 std::lock_guard<std::mutex> its_lock(mutex_);
181
182 if(endpoint_impl<Protocol>::sending_blocked_) {
183 return false;
184 }
185
186 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
187 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
188 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
189 _data[VSOMEIP_CLIENT_POS_MIN], _data[VSOMEIP_CLIENT_POS_MAX]);
190 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
191 _data[VSOMEIP_SESSION_POS_MIN], _data[VSOMEIP_SESSION_POS_MAX]);
192
193 clients_mutex_.lock();
194 auto found_client = clients_.find(its_client);
195 if (found_client != clients_.end()) {
196 auto found_session = found_client->second.find(its_session);
197 if (found_session != found_client->second.end()) {
198 its_target = found_session->second;
199 is_valid_target = true;
200 found_client->second.erase(its_session);
201 } else {
202 VSOMEIP_WARNING << "server_endpoint::send: session_id 0x"
203 << std::hex << its_session
204 << " not found for client 0x" << its_client;
205 const method_t its_method =
206 VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
207 _data[VSOMEIP_METHOD_POS_MAX]);
208 if (its_service == VSOMEIP_SD_SERVICE
209 && its_method == VSOMEIP_SD_METHOD) {
210 VSOMEIP_ERROR << "Clearing clients map as a request was "
211 "received on SD port";
212 clients_.clear();
213 is_valid_target = get_default_target(its_service, its_target);
214 }
215 }
216 } else {
217 is_valid_target = get_default_target(its_service, its_target);
218 }
219 clients_mutex_.unlock();
220
221 if (is_valid_target) {
222 is_valid_target = send_intern(its_target, _data, _size);
223 }
224 }
225 return is_valid_target;
226}

238template<typename Protocol>
239bool server_endpoint_impl<Protocol>::send_intern(
240 endpoint_type _target, const byte_t *_data, uint32_t _size) {
241
242 switch (check_message_size(_data, _size, _target)) {
243 case endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT:
244 return true;
245 break;
246 case endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG:
247 return false;
248 break;
249 case endpoint_impl<Protocol>::cms_ret_e::MSG_OK:
250 default:
251 break;
252 }
253 if (!prepare_stop_handlers_.empty()) {
254 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
255 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
256 if (prepare_stop_handlers_.find(its_service) != prepare_stop_handlers_.end()) {
257 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
258 _data[VSOMEIP_METHOD_POS_MIN], _data[VSOMEIP_METHOD_POS_MAX]);
259 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
260 _data[VSOMEIP_CLIENT_POS_MIN], _data[VSOMEIP_CLIENT_POS_MAX]);
261 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
262 _data[VSOMEIP_SESSION_POS_MIN], _data[VSOMEIP_SESSION_POS_MAX]);
263 VSOMEIP_WARNING << "server_endpoint::send: Service is stopping, ignoring message: ["
264 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
265 << std::hex << std::setw(4) << std::setfill('0') << its_method << "."
266 << std::hex << std::setw(4) << std::setfill('0') << its_client << "."
267 << std::hex << std::setw(4) << std::setfill('0') << its_session << "]";
268 return false;
269 }
270 }
271
272 const queue_iterator_type target_queue_iterator = find_or_create_queue_unlocked(_target);
273
274 bool must_depart(false);
275
276#if 0
277 std::stringstream msg;
278 msg << "sei::send_intern: ";
279 for (uint32_t i = 0; i < _size; i++)
280 msg << std::hex << std::setw(2) << std::setfill('0') << (int)_data[i] << " ";
281 VSOMEIP_DEBUG << msg.str();
282#endif
283 // STEP 1: determine the correct train
284 std::shared_ptr<train> target_train = find_or_create_train_unlocked(_target);
285
286 const bool queue_size_zero_on_entry(target_queue_iterator->second.second.empty());
287 if (!check_queue_limit(_data, _size, target_queue_iterator->second.first)) {
288 return false;
289 }
290 // STEP 2: Determine elapsed time and update the departure time and cancel the timer
291 target_train->update_departure_time_and_stop_departure();
292
293 // STEP 3: Get configured timings
294 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
295 _data[VSOMEIP_SERVICE_POS_MIN], _data[VSOMEIP_SERVICE_POS_MAX]);
296 const method_t its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
297 _data[VSOMEIP_METHOD_POS_MAX]);
298
299 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
300 if (its_service != VSOMEIP_SD_SERVICE && its_method != VSOMEIP_SD_METHOD) {
301 get_configured_times_from_endpoint(its_service, its_method,
302 &its_debouncing, &its_retention);
303 }
304
305 // STEP 4: Check if the passenger enters an empty train
306 const std::pair<service_t, method_t> its_identifier = std::make_pair(
307 its_service, its_method);
308 if (target_train->passengers_.empty()) {
309 target_train->departure_ = its_retention;
310 } else {
311 if (target_train->passengers_.end()
312 != target_train->passengers_.find(its_identifier)) {
313 must_depart = true;
314 } else {
315 // STEP 5: Check whether the current message fits into the current train
316 if (target_train->buffer_->size() + _size > endpoint_impl<Protocol>::max_message_size_) {
317 must_depart = true;
318 } else {
319 // STEP 6: Check debouncing time
320 if (its_debouncing > target_train->minimal_max_retention_time_) {
321 // train's latest departure would already undershot new
322 // passenger's debounce time
323 must_depart = true;
324 } else {
325 if (its_debouncing > target_train->departure_) {
326 // train departs earlier as the new passenger's debounce
327 // time allows
328 must_depart = true;
329 } else {
330 // STEP 7: Check maximum retention time
331 if (its_retention < target_train->minimal_debounce_time_) {
332 // train's earliest departure would already exceed
333 // the new passenger's retention time.
334 must_depart = true;
335 } else {
336 if (its_retention < target_train->departure_) {
337 target_train->departure_ = its_retention;
338 }
339 }
340 }
341 }
342 }
343 }
344 }
345
346 // STEP 8: if necessary, send current buffer and create a new one
347 if (must_depart) {
348 // STEP 8.1: check if debounce time would be undershot here if the train
349 // departs. Block sending until train is allowed to depart.
350 wait_until_debounce_time_reached(target_train);
351 queue_train(target_queue_iterator, target_train,
352 queue_size_zero_on_entry);
353 target_train->departure_ = its_retention;
354 }
355
356 // STEP 9: insert current message buffer
357 target_train->buffer_->insert(target_train->buffer_->end(), _data, _data + _size);
358 target_train->passengers_.insert(its_identifier);
359 // STEP 9.1: update the trains minimal debounce time if necessary
360 if (its_debouncing < target_train->minimal_debounce_time_) {
361 target_train->minimal_debounce_time_ = its_debouncing;
362 }
363 // STEP 9.2: update the trains minimal maximum retention time if necessary
364 if (its_retention < target_train->minimal_max_retention_time_) {
365 target_train->minimal_max_retention_time_ = its_retention;
366 }
367
368 // STEP 10: restart timer with current departure time
369#ifndef _WIN32
370 target_train->departure_timer_->expires_from_now(target_train->departure_);
371#else
372 target_train->departure_timer_->expires_from_now(
373 std::chrono::duration_cast<
374 std::chrono::steady_clock::duration>(target_train->departure_));
375#endif
376 target_train->departure_timer_->async_wait(
377 std::bind(&server_endpoint_impl<Protocol>::flush_cbk,
378 this->shared_from_this(), _target,
379 target_train, std::placeholders::_1));
380
381 return (true);
382}

384template<typename Protocol>
385void server_endpoint_impl<Protocol>::send_segments(
386 const tp::tp_split_messages_t &_segments, const endpoint_type &_target) {
387
388 if (_segments.size() == 0)
389 return;
390
391 const queue_iterator_type target_queue_iterator = find_or_create_queue_unlocked(_target);
392 const bool queue_size_zero_on_entry(target_queue_iterator->second.second.empty());
393
394 std::shared_ptr<train> target_train = find_or_create_train_unlocked(_target);
395 target_train->update_departure_time_and_stop_departure();
396
397 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
398 (*(_segments[0]))[VSOMEIP_SERVICE_POS_MIN], (*(_segments[0]))[VSOMEIP_SERVICE_POS_MAX]);
399 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
400 (*(_segments[0]))[VSOMEIP_METHOD_POS_MIN], (*(_segments[0]))[VSOMEIP_METHOD_POS_MAX]);
401
402 std::chrono::nanoseconds its_debouncing(0), its_retention(0);
403 if (its_service != VSOMEIP_SD_SERVICE && its_method != VSOMEIP_SD_METHOD) {
404 get_configured_times_from_endpoint(its_service, its_method,
405 &its_debouncing, &its_retention);
406 }
407 // update the trains minimal debounce time if necessary
408 if (its_debouncing < target_train->minimal_debounce_time_) {
409 target_train->minimal_debounce_time_ = its_debouncing;
410 }
411 // update the trains minimal maximum retention time if necessary
412 if (its_retention < target_train->minimal_max_retention_time_) {
413 target_train->minimal_max_retention_time_ = its_retention;
414 }
415 // We only need to respect the debouncing. There is no need to wait for further
416 // messages as we will send several now anyway.
417 if (!target_train->passengers_.empty()) {
418 wait_until_debounce_time_reached(target_train);
419 queue_train(target_queue_iterator, target_train, queue_size_zero_on_entry);
420 }
421
422 const bool queue_size_still_zero(target_queue_iterator->second.second.empty());
423 for (const auto &s : _segments) {
424 target_queue_iterator->second.second.emplace_back(s);
425 target_queue_iterator->second.first += s->size();
426 }
427 if (queue_size_still_zero && !target_queue_iterator->second.second.empty()) { // no writing in progress
428 // respect minimal debounce time
429 wait_until_debounce_time_reached(target_train);
430 // ignore retention time and send immediately as the train is full anyway
431 send_queued(target_queue_iterator);
432 }
433 target_train->last_departure_ = std::chrono::steady_clock::now();
434}

436template<typename Protocol>
437void server_endpoint_impl<Protocol>::wait_until_debounce_time_reached(
438 const std::shared_ptr<train>& _train) const {
439 const std::chrono::nanoseconds time_since_last_departure =
440 std::chrono::duration_cast<std::chrono::nanoseconds>(
441 std::chrono::steady_clock::now() - _train->last_departure_);
442
443 if (time_since_last_departure < _train->minimal_debounce_time_) {
444 std::this_thread::sleep_for(
445 _train->minimal_debounce_time_ - time_since_last_departure);
446 }
447}

450template<typename Protocol>
451typename endpoint_impl<Protocol>::cms_ret_e server_endpoint_impl<Protocol>::check_message_size(
452 const std::uint8_t * const _data, std::uint32_t _size,
453 const endpoint_type& _target) {
454 typename endpoint_impl<Protocol>::cms_ret_e ret(endpoint_impl<Protocol>::cms_ret_e::MSG_OK);
455 if (endpoint_impl<Protocol>::max_message_size_ != MESSAGE_SIZE_UNLIMITED
456 && _size > endpoint_impl<Protocol>::max_message_size_) {
457 if (endpoint_impl<Protocol>::is_supporting_someip_tp_ && _data != nullptr) {
458 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
459 _data[VSOMEIP_SERVICE_POS_MIN],
460 _data[VSOMEIP_SERVICE_POS_MAX]);
461 const method_t its_method = VSOMEIP_BYTES_TO_WORD(
462 _data[VSOMEIP_METHOD_POS_MIN],
463 _data[VSOMEIP_METHOD_POS_MAX]);
464 if (tp_segmentation_enabled(its_service, its_method)) {
465 send_segments(tp::tp::tp_split_message(_data, _size), _target);
466 return endpoint_impl<Protocol>::cms_ret_e::MSG_WAS_SPLIT;
467 }
468 }
469 VSOMEIP_ERROR << "sei::send_intern: Dropping too big message (" << _size
470 << " Bytes). Maximum allowed message size is: "
471 << endpoint_impl<Protocol>::max_message_size_ << " Bytes.";
472 ret = endpoint_impl<Protocol>::cms_ret_e::MSG_TOO_BIG;
473 }
474 return ret;
475}

477template<typename Protocol>
478bool server_endpoint_impl<Protocol>::check_queue_limit(const uint8_t *_data, std::uint32_t _size,
479 std::size_t _current_queue_size) const {
480 if (endpoint_impl<Protocol>::queue_limit_ != QUEUE_SIZE_UNLIMITED
481 && _current_queue_size + _size
482 > endpoint_impl<Protocol>::queue_limit_) {
483 service_t its_service(0);
484 method_t its_method(0);
485 client_t its_client(0);
486 session_t its_session(0);
487 if (_size >= VSOMEIP_SESSION_POS_MAX) {
488 // this will yield wrong IDs for local communication as the commands
489 // are prepended to the actual payload
490 // it will print:
491 // (lowbyte service ID + highbyte methoid)
492 // [(Command + lowerbyte sender's client ID).
493 // highbyte sender's client ID + lowbyte command size.
494 // lowbyte methodid + highbyte vsomeip length]
495 its_service = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SERVICE_POS_MIN],
496 _data[VSOMEIP_SERVICE_POS_MAX]);
497 its_method = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_METHOD_POS_MIN],
498 _data[VSOMEIP_METHOD_POS_MAX]);
499 its_client = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_CLIENT_POS_MIN],
500 _data[VSOMEIP_CLIENT_POS_MAX]);
501 its_session = VSOMEIP_BYTES_TO_WORD(_data[VSOMEIP_SESSION_POS_MIN],
502 _data[VSOMEIP_SESSION_POS_MAX]);
503 }
504 VSOMEIP_ERROR << "sei::send_intern: queue size limit (" << std::dec
505 << endpoint_impl<Protocol>::queue_limit_
506 << ") reached. Dropping message ("
507 << std::hex << std::setw(4) << std::setfill('0') << its_client <<"): ["
508 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
509 << std::hex << std::setw(4) << std::setfill('0') << its_method << "."
510 << std::hex << std::setw(4) << std::setfill('0') << its_session << "]"
511 << " queue_size: " << std::dec << _current_queue_size
512 << " data size: " << std::dec << _size;
513 return false;
514 }
515 return true;
516}

518template<typename Protocol>
519void server_endpoint_impl<Protocol>::queue_train(
520 const queue_iterator_type _queue_iterator,
521 const std::shared_ptr<train>& _train,
522 bool _queue_size_zero_on_entry) {
523 _queue_iterator->second.second.emplace_back(_train->buffer_);
524 _queue_iterator->second.first += _train->buffer_->size();
525 _train->last_departure_ = std::chrono::steady_clock::now();
526 _train->passengers_.clear();
527 _train->buffer_ = std::make_shared<message_buffer_t>();
528 _train->minimal_debounce_time_ = std::chrono::nanoseconds::max();
529 _train->minimal_max_retention_time_ = std::chrono::nanoseconds::max();
530 if (_queue_size_zero_on_entry && !_queue_iterator->second.second.empty()) { // no writing in progress
531 send_queued(_queue_iterator);
532 }
533}

535template<typename Protocol>
536typename server_endpoint_impl<Protocol>::queue_iterator_type
537server_endpoint_impl<Protocol>::find_or_create_queue_unlocked(const endpoint_type& _target) {
538 queue_iterator_type target_queue_iterator = queues_.find(_target);
539 if (target_queue_iterator == queues_.end()) {
540 target_queue_iterator = queues_.insert(queues_.begin(),
541 std::make_pair(
542 _target,
543 std::make_pair(std::size_t(0),
544 std::deque<message_buffer_ptr_t>())
545 ));
546 }
547 return target_queue_iterator;
548}

550template<typename Protocol>
551std::shared_ptr<train> server_endpoint_impl<Protocol>::find_or_create_train_unlocked(
552 const endpoint_type& _target) {
553 auto train_iter = trains_.find(_target);
554 if (train_iter == trains_.end()) {
555 train_iter = trains_.insert(trains_.begin(),
556 std::make_pair(_target, std::make_shared<train>(this->service_)));
557 }
558 return train_iter->second;
559}

561template<typename Protocol>
562bool server_endpoint_impl<Protocol>::flush(
563 endpoint_type _target,
564 const std::shared_ptr<train>& _train) {
565 std::lock_guard<std::mutex> its_lock(mutex_);
566 bool is_flushed = false;
567 if (!_train->buffer_->empty()) {
568 const queue_iterator_type target_queue_iterator = queues_.find(_target);
569 if (target_queue_iterator != queues_.end()) {
570 const bool queue_size_zero_on_entry(target_queue_iterator->second.second.empty());
571 queue_train(target_queue_iterator, _train, queue_size_zero_on_entry);
572 is_flushed = true;
573 } else {
574 std::stringstream ss;
575 ss << "sei::flush couldn't find target queue, won't queue train to: "
576 << get_remote_information(_target) << " passengers: ";
577 for (const auto& p : _train->passengers_) {
578 ss << "[" << std::hex << std::setw(4) << std::setfill('0')
579 << p.first << ":" << p.second << "] ";
580 }
581 VSOMEIP_WARNING << ss.str();
582 }
583 }
584 return is_flushed;
585}

593template<typename Protocol>
594void server_endpoint_impl<Protocol>::send_cbk(
595 const queue_iterator_type _queue_iterator,
596 boost::system::error_code const &_error, std::size_t _bytes) {
597 (void)_bytes;
598
599 {
600 std::lock_guard<std::mutex> its_sent_lock(sent_mutex_);
601 is_sending_ = false;
602
603 boost::system::error_code ec;
604 sent_timer_.cancel(ec);
605 }
606
607 std::lock_guard<std::mutex> its_lock(mutex_);
608
609 auto check_if_all_msgs_for_stopped_service_are_sent = [&]() {
610 bool found_service_msg(false);
611 service_t its_stopped_service(ANY_SERVICE);
612 for (auto stp_hndlr_iter = prepare_stop_handlers_.begin();
613 stp_hndlr_iter != prepare_stop_handlers_.end();) {
614 its_stopped_service = stp_hndlr_iter->first;
615 if (its_stopped_service == ANY_SERVICE) {
616 ++stp_hndlr_iter;
617 continue;
618 }
619 for (const auto& q : queues_) {
620 for (const auto& msg : q.second.second ) {
621 const service_t its_service = VSOMEIP_BYTES_TO_WORD(
622 (*msg)[VSOMEIP_SERVICE_POS_MIN],
623 (*msg)[VSOMEIP_SERVICE_POS_MAX]);
624 if (its_service == its_stopped_service) {
625 found_service_msg = true;
626 break;
627 }
628 }
629 if (found_service_msg) {
630 break;
631 }
632 }
633 if (found_service_msg) {
634 ++stp_hndlr_iter;
635 found_service_msg = false;
636 } else { // all messages of the to be stopped service have been sent
637 auto handler = stp_hndlr_iter->second;
638 auto ptr = this->shared_from_this();
639 #ifndef _WIN32
640 endpoint_impl<Protocol>::
641 #endif
642 service_.post([ptr, handler, its_stopped_service](){
643 handler(ptr, its_stopped_service);
644 });
645 stp_hndlr_iter = prepare_stop_handlers_.erase(stp_hndlr_iter);
646 }
647 }
648 };
649
650 auto check_if_all_queues_are_empty = [&](){
651 if (prepare_stop_handlers_.size() > 1) {
652 // before the endpoint was stopped completely other
653 // prepare_stop_handlers have been queued ensure to call them as well
654 check_if_all_msgs_for_stopped_service_are_sent();
655 }
656 if (std::all_of(queues_.begin(), queues_.end(), [&]
657 #ifndef _WIN32
658 (const typename queue_type::value_type& q)
659 #else
660 (const std::pair<endpoint_type,std::pair<size_t, std::deque<message_buffer_ptr_t>>>& q)
661 #endif
662 { return q.second.second.empty(); })) {
663 // all outstanding response have been sent.
664 auto found_cbk = prepare_stop_handlers_.find(ANY_SERVICE);
665 if (found_cbk != prepare_stop_handlers_.end()) {
666 auto handler = found_cbk->second;
667 auto ptr = this->shared_from_this();
668 #ifndef _WIN32
669 endpoint_impl<Protocol>::
670 #endif
671 service_.post([ptr, handler](){
672 handler(ptr, ANY_SERVICE);
673 });
674 prepare_stop_handlers_.erase(found_cbk);
675 }
676 }
677 };
678
679 auto& its_qpair = _queue_iterator->second;
680 if (!_error) {
681 its_qpair.first -= its_qpair.second.front()->size();
682 its_qpair.second.pop_front();
683
684 if (!prepare_stop_handlers_.empty() && !endpoint_impl<Protocol>::sending_blocked_) {
685 // only one service instance is stopped
686 check_if_all_msgs_for_stopped_service_are_sent();
687 }
688
689 if (its_qpair.second.size() > 0) {
690 send_queued(_queue_iterator);
691 } else if (!prepare_stop_handlers_.empty() && endpoint_impl<Protocol>::sending_blocked_) {
692 // endpoint is shutting down completely
693 queues_.erase(_queue_iterator);
694 check_if_all_queues_are_empty();
695 }
696 } else {
697 message_buffer_ptr_t its_buffer;
698 if (_queue_iterator->second.second.size()) {
699 its_buffer = _queue_iterator->second.second.front();
700 }
701 service_t its_service(0);
702 method_t its_method(0);
703 client_t its_client(0);
704 session_t its_session(0);
705 if (its_buffer && its_buffer->size() > VSOMEIP_SESSION_POS_MAX) {
706 its_service = VSOMEIP_BYTES_TO_WORD(
707 (*its_buffer)[VSOMEIP_SERVICE_POS_MIN],
708 (*its_buffer)[VSOMEIP_SERVICE_POS_MAX]);
709 its_method = VSOMEIP_BYTES_TO_WORD(
710 (*its_buffer)[VSOMEIP_METHOD_POS_MIN],
711 (*its_buffer)[VSOMEIP_METHOD_POS_MAX]);
712 its_client = VSOMEIP_BYTES_TO_WORD(
713 (*its_buffer)[VSOMEIP_CLIENT_POS_MIN],
714 (*its_buffer)[VSOMEIP_CLIENT_POS_MAX]);
715 its_session = VSOMEIP_BYTES_TO_WORD(
716 (*its_buffer)[VSOMEIP_SESSION_POS_MIN],
717 (*its_buffer)[VSOMEIP_SESSION_POS_MAX]);
718 }
719 // error: sending of outstanding responses isn't started again
720 // delete remaining outstanding responses
721 VSOMEIP_WARNING << "sei::send_cbk received error: " << _error.message()
722 << " (" << std::dec << _error.value() << ") "
723 << get_remote_information(_queue_iterator) << " "
724 << std::dec << _queue_iterator->second.second.size() << " "
725 << std::dec << _queue_iterator->second.first << " ("
726 << std::hex << std::setw(4) << std::setfill('0') << its_client <<"): ["
727 << std::hex << std::setw(4) << std::setfill('0') << its_service << "."
728 << std::hex << std::setw(4) << std::setfill('0') << its_method << "."
729 << std::hex << std::setw(4) << std::setfill('0') << its_session << "]";
730 queues_.erase(_queue_iterator);
731 if (!prepare_stop_handlers_.empty()) {
732 if (endpoint_impl<Protocol>::sending_blocked_) {
733 // endpoint is shutting down completely, ensure to call
734 // prepare_stop_handlers even in error cases
735 check_if_all_queues_are_empty();
736 } else {
737 // only one service instance is stopped
738 check_if_all_msgs_for_stopped_service_are_sent();
739 }
740 }
741
742 }
743}
