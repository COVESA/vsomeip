158 basic_socket_acceptor_ext(boost::asio::io_context& io_context,
159 const endpoint_type& endpoint, bool reuse_addr = true)
160 : basic_io_object<BOOST_ASIO_SVC_T>(io_context)
161 {
162 boost::system::error_code ec;
163 const protocol_type protocol = endpoint.protocol();
164 this->get_service().open(this->get_implementation(), protocol, ec);
165 boost::asio::detail::throw_error(ec, "open");
166 if (reuse_addr)
167 {
168 this->get_service().set_option(this->get_implementation(),
169 socket_base::reuse_address(true), ec);
170 boost::asio::detail::throw_error(ec, "set_option");
171 }
172 this->get_service().bind(this->get_implementation(), endpoint, ec);
173 boost::asio::detail::throw_error(ec, "bind");
174 this->get_service().listen(this->get_implementation(),
175 socket_base::max_listen_connections, ec);
176 boost::asio::detail::throw_error(ec, "listen");
177 }
