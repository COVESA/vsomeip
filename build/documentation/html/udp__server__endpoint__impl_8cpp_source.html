28udp_server_endpoint_impl::udp_server_endpoint_impl(
29 const std::shared_ptr<endpoint_host>& _endpoint_host,
30 const std::shared_ptr<routing_host>& _routing_host,
31 const endpoint_type& _local,
32 boost::asio::io_service &_io,
33 const std::shared_ptr<configuration>& _configuration) :
34 server_endpoint_impl<ip::udp_ext>(_endpoint_host, _routing_host, _local,
35 _io, VSOMEIP_MAX_UDP_MESSAGE_SIZE,
36 _configuration->get_endpoint_queue_limit(_configuration->get_unicast_address().to_string(), _local.port()),
37 _configuration),
38 unicast_socket_(_io, _local.protocol()),
39 unicast_recv_buffer_(VSOMEIP_MAX_UDP_MESSAGE_SIZE, 0),
40 multicast_id_(0),
41 joined_group_(false),
42 local_port_(_local.port()),
43 tp_reassembler_(std::make_shared<tp::tp_reassembler>(_configuration->get_max_message_size_unreliable(), _io)),
44 tp_cleanup_timer_(_io) {
45 is_supporting_someip_tp_ = true;
46
47 boost::system::error_code ec;
48
49 boost::asio::socket_base::reuse_address optionReuseAddress(true);
50 unicast_socket_.set_option(optionReuseAddress, ec);
51 boost::asio::detail::throw_error(ec, "reuse address");
52
53#ifndef _WIN32
54 // If specified, bind to device
55 std::string its_device(configuration_->get_device());
56 if (its_device != "") {
57 if (setsockopt(unicast_socket_.native_handle(),
58 SOL_SOCKET, SO_BINDTODEVICE, its_device.c_str(), (int)its_device.size()) == -1) {
59 VSOMEIP_WARNING << "UDP Server: Could not bind to device \"" << its_device << "\"";
60 }
61 }
62#endif
63
64 unicast_socket_.bind(_local, ec);
65 boost::asio::detail::throw_error(ec, "bind");
66
67 if (local_.address().is_v4()) {
68 boost::asio::ip::multicast::outbound_interface option(_local.address().to_v4());
69 unicast_socket_.set_option(option, ec);
70 boost::asio::detail::throw_error(ec, "outbound interface option IPv4");
71 } else if (local_.address().is_v6()) {
72 boost::asio::ip::multicast::outbound_interface option(
73 static_cast<unsigned int>(local_.address().to_v6().scope_id()));
74 unicast_socket_.set_option(option, ec);
75 boost::asio::detail::throw_error(ec, "outbound interface option IPv6");
76 }
77
78 boost::asio::socket_base::broadcast option(true);
79 unicast_socket_.set_option(option, ec);
80 boost::asio::detail::throw_error(ec, "broadcast option");
81
82 const std::uint32_t its_udp_recv_buffer_size =
83 configuration_->get_udp_receive_buffer_size();
84 unicast_socket_.set_option(boost::asio::socket_base::receive_buffer_size(
85 its_udp_recv_buffer_size), ec);
86
87 if (ec) {
88 VSOMEIP_WARNING << "udp_server_endpoint_impl:: couldn't set "
89 << "SO_RCVBUF: " << ec.message() << " to: " << std::dec
90 << its_udp_recv_buffer_size << " local port: " << std::dec
91 << local_port_;
92 } else {
93 boost::asio::socket_base::receive_buffer_size its_option;
94 unicast_socket_.get_option(its_option, ec);
95 if (ec) {
96 VSOMEIP_WARNING << "udp_server_endpoint_impl: couldn't get "
97 << "SO_RCVBUF: " << ec.message() << " local port:"
98 << std::dec << local_port_;
99 } else {
100 VSOMEIP_INFO << "udp_server_endpoint_impl: SO_RCVBUF is: "
101 << std::dec << its_option.value();
102 }
103 }
104
105#ifdef _WIN32
106 const char* optval("0001");
107 ::setsockopt(unicast_socket_.native_handle(), IPPROTO_IP, IP_PKTINFO,
108 optval, sizeof(optval));
109#else
110 int optval(1);
111 ::setsockopt(unicast_socket_.native_handle(), IPPROTO_IP, IP_PKTINFO,
112 &optval, sizeof(optval));
113#endif
114}

207bool udp_server_endpoint_impl::send_error(
208 const std::shared_ptr<endpoint_definition> _target,
209 const byte_t *_data, uint32_t _size) {
210
211 bool ret(false);
212 std::lock_guard<std::mutex> its_lock(mutex_);
213 const endpoint_type its_target(_target->get_address(), _target->get_port());
214 const queue_iterator_type target_queue_iterator(find_or_create_queue_unlocked(its_target));
215 auto& its_qpair = target_queue_iterator->second;
216 const bool queue_size_zero_on_entry(its_qpair.second.empty());
217
218 if (check_message_size(nullptr, _size, its_target) == endpoint_impl::cms_ret_e::MSG_OK &&
219 check_queue_limit(_data, _size, its_qpair.first)) {
220 its_qpair.second.emplace_back(
221 std::make_shared<message_buffer_t>(_data, _data + _size));
222 its_qpair.first += _size;
223
224 if (queue_size_zero_on_entry) { // no writing in progress
225 send_queued(target_queue_iterator);
226 }
227 ret = true;
228 }
229 return ret;
230}

298void udp_server_endpoint_impl::join_unlocked(const std::string &_address) {
299
300 bool has_received(false);
301
302 //
303 // join_func must be called with multicast_mutex_ being hold!
304 //
305 auto join_func = [this](const std::string &_address) {
306 try {
307 VSOMEIP_DEBUG << "Joining to multicast group " << _address
308 << " from " << local_.address().to_string();
309
310 boost::system::error_code ec;
311
312 bool is_v4(false);
313 bool is_v6(false);
314 {
315 std::lock_guard<std::mutex> its_lock(local_mutex_);
316 is_v4 = local_.address().is_v4();
317 is_v6 = local_.address().is_v6();
318 }
319
320 if (multicast_recv_buffer_.empty())
321 multicast_recv_buffer_.resize(VSOMEIP_MAX_UDP_MESSAGE_SIZE, 0);
322
323 if (!multicast_local_) {
324 if (is_v4) {
325 multicast_local_ = std::unique_ptr<endpoint_type>(
326 new endpoint_type(boost::asio::ip::address_v4::any(), local_port_));
327 }
328 if (is_v6) {
329 multicast_local_ = std::unique_ptr<endpoint_type>(
330 new endpoint_type(boost::asio::ip::address_v6::any(), local_port_));
331 }
332 }
333
334 if (!multicast_socket_) {
335 multicast_socket_ = std::unique_ptr<socket_type>(
336 new socket_type(service_, local_.protocol()));
337
338 boost::asio::socket_base::reuse_address optionReuseAddress(true);
339 multicast_socket_->set_option(optionReuseAddress, ec);
340 boost::asio::detail::throw_error(ec, "reuse address in multicast");
341 boost::asio::socket_base::broadcast optionBroadcast(true);
342 multicast_socket_->set_option(optionBroadcast, ec);
343 boost::asio::detail::throw_error(ec, "set broadcast option");
344
345 multicast_socket_->bind(*multicast_local_, ec);
346 boost::asio::detail::throw_error(ec, "bind multicast");
347
348 const std::uint32_t its_udp_recv_buffer_size =
349 configuration_->get_udp_receive_buffer_size();
350
351 multicast_socket_->set_option(boost::asio::socket_base::receive_buffer_size(
352 its_udp_recv_buffer_size), ec);
353
354 if (ec) {
355 VSOMEIP_WARNING << "udp_server_endpoint_impl:: couldn't set "
356 << "SO_RCVBUF: " << ec.message() << " to: " << std::dec
357 << its_udp_recv_buffer_size << " local port: " << std::dec
358 << local_port_;
359 } else {
360 boost::asio::socket_base::receive_buffer_size its_option;
361 multicast_socket_->get_option(its_option, ec);
362
363 if (ec) {
364 VSOMEIP_WARNING << "udp_server_endpoint_impl: couldn't get "
365 << "SO_RCVBUF: " << ec.message() << " local port:"
366 << std::dec << local_port_;
367 } else {
368 VSOMEIP_INFO << "udp_server_endpoint_impl: SO_RCVBUF (Multicast) is: "
369 << std::dec << its_option.value();
370 }
371 }
372
373#ifdef _WIN32
374 const char* optval("0001");
375 if (is_v4) {
376 ::setsockopt(multicast_socket_->native_handle(), IPPROTO_IP, IP_PKTINFO,
377 optval, sizeof(optval));
378 } else if (is_v6) {
379 ::setsockopt(multicast_socket_->native_handle(), IPPROTO_IPV6, IPV6_PKTINFO,
380 optval, sizeof(optval));
381 }
382#else
383 int optval(1);
384 if (is_v4) {
385 ::setsockopt(multicast_socket_->native_handle(), IPPROTO_IP, IP_PKTINFO,
386 &optval, sizeof(optval));
387 } else {
388 ::setsockopt(multicast_socket_->native_handle(), IPPROTO_IPV6, IPV6_RECVPKTINFO,
389 &optval, sizeof(optval));
390 }
391#endif
392 multicast_id_++;
393 receive_multicast(multicast_id_);
394 }
395
396 if (is_v4) {
397 multicast_socket_->set_option(ip::udp_ext::socket::reuse_address(true));
398 multicast_socket_->set_option(
399 boost::asio::ip::multicast::enable_loopback(false));
400 multicast_socket_->set_option(boost::asio::ip::multicast::join_group(
401 boost::asio::ip::address::from_string(_address).to_v4(),
402 local_.address().to_v4()));
403 } else if (is_v6) {
404 multicast_socket_->set_option(ip::udp_ext::socket::reuse_address(true));
405 multicast_socket_->set_option(
406 boost::asio::ip::multicast::enable_loopback(false));
407 multicast_socket_->set_option(boost::asio::ip::multicast::join_group(
408 boost::asio::ip::address::from_string(_address).to_v6(),
409 local_.address().to_v6().scope_id()));
410 }
411
412 joined_[_address] = false;
413 joined_group_ = true;
414
415 } catch (const std::exception &e) {
416 VSOMEIP_ERROR << "udp_server_endpoint_impl::join" << ":" << e.what();
417 }
418 };
419
420 if (!is_joined(_address, &has_received)) {
421 join_func(_address);
422 } else if (!has_received) {
423 // joined the multicast group but didn't receive a event yet -> rejoin
424 leave_unlocked(_address);
425 join_func(_address);
426 }
427}

435void udp_server_endpoint_impl::leave_unlocked(const std::string &_address) {
436
437 try {
438 if (is_joined(_address)) {
439 VSOMEIP_DEBUG << "Leaving the multicast group " << _address
440 << " from " << local_.address().to_string();
441
442 bool is_v4(false);
443 bool is_v6(false);
444 {
445 std::lock_guard<std::mutex> its_lock(local_mutex_);
446 is_v4 = local_.address().is_v4();
447 is_v6 = local_.address().is_v6();
448 }
449 if (is_v4) {
450 multicast_socket_->set_option(boost::asio::ip::multicast::leave_group(
451 boost::asio::ip::address::from_string(_address)));
452 } else if (is_v6) {
453 multicast_socket_->set_option(boost::asio::ip::multicast::leave_group(
454 boost::asio::ip::address::from_string(_address)));
455 }
456
457 joined_.erase(_address);
458 if (0 == joined_.size()) {
459 joined_group_ = false;
460
461 boost::system::error_code ec;
462 multicast_socket_->cancel(ec);
463
464 multicast_socket_.reset(nullptr);
465 multicast_local_.reset(nullptr);
466 }
467 }
468 }
469 catch (const std::exception &e) {
470 VSOMEIP_ERROR << __func__ << ":" << e.what();
471 }
472}

503void udp_server_endpoint_impl::on_unicast_received(
504 boost::system::error_code const &_error,
505 std::size_t _bytes,
506 boost::asio::ip::address const &_destination) {
507
508 if (_error != boost::asio::error::operation_aborted) {
509 {
510 // By locking the multicast mutex here it is ensured that unicast
511 // & multicast messages are not processed in parallel. This aligns
512 // the behavior of endpoints with one and two active sockets.
513 std::lock_guard<std::mutex> its_lock(multicast_mutex_);
514 on_message_received(_error, _bytes, _destination,
515 unicast_remote_, unicast_recv_buffer_);
516 }
517 receive_unicast();
518 }
519}
520
521void udp_server_endpoint_impl::on_multicast_received(
522 boost::system::error_code const &_error,
523 std::size_t _bytes,
524 boost::asio::ip::address const &_destination,
525 uint8_t _multicast_id) {
526
527 std::lock_guard<std::mutex> its_lock(multicast_mutex_);
528 if (_error != boost::asio::error::operation_aborted) {
529 // Filter messages sent from the same source address
530 if (multicast_remote_.address() != local_.address()) {
531 on_message_received(_error, _bytes, _destination,
532 multicast_remote_, multicast_recv_buffer_);
533 }
534
535 receive_multicast(_multicast_id);
536 }
537}

539void udp_server_endpoint_impl::on_message_received(
540 boost::system::error_code const &_error, std::size_t _bytes,
541 boost::asio::ip::address const &_destination,
542 endpoint_type const &_remote,
543 message_buffer_t const &_buffer) {
544#if 0
545 std::stringstream msg;
546 msg << "usei::rcb(" << _error.message() << "): ";
547 for (std::size_t i = 0; i < _bytes; ++i)
548 msg << std::hex << std::setw(2) << std::setfill('0')
549 << (int) recv_buffer_[i] << " ";
550 VSOMEIP_INFO << msg.str();
551#endif
552 std::shared_ptr<routing_host> its_host = routing_host_.lock();
553
554 if (its_host) {
555 if (!_error && 0 < _bytes) {
556 std::size_t remaining_bytes = _bytes;
557 std::size_t i = 0;
558 const boost::asio::ip::address its_remote_address(_remote.address());
559 const std::uint16_t its_remote_port(_remote.port());
560 do {
561 uint64_t read_message_size
562 = utility::get_message_size(&_buffer[i],
563 remaining_bytes);
564 if (read_message_size > MESSAGE_SIZE_UNLIMITED) {
565 VSOMEIP_ERROR << "Message size exceeds allowed maximum!";
566 return;
567 }
568 uint32_t current_message_size = static_cast<uint32_t>(read_message_size);
569 if (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
570 current_message_size <= remaining_bytes) {
571 if (remaining_bytes - current_message_size > remaining_bytes) {
572 VSOMEIP_ERROR << "buffer underflow in udp client endpoint ~> abort!";
573 return;
574 } else if (current_message_size > VSOMEIP_RETURN_CODE_POS &&
575 (_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION ||
576 !utility::is_valid_message_type(tp::tp::tp_flag_unset(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) ||
578 (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS]) && get_local_port() == configuration_->get_sd_port())
579 )) {
580 if (_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS] != VSOMEIP_PROTOCOL_VERSION) {
581 VSOMEIP_ERROR << "use: Wrong protocol version: 0x"
582 << std::hex << std::setw(2) << std::setfill('0')
583 << std::uint32_t(_buffer[i + VSOMEIP_PROTOCOL_VERSION_POS])
584 << " local: " << get_address_port_local()
585 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
586 // ensure to send back a message w/ wrong protocol version
587 its_host->on_message(&_buffer[i],
588 VSOMEIP_SOMEIP_HEADER_SIZE + 8, this,
589 _destination,
590 VSOMEIP_ROUTING_CLIENT,
591 std::make_pair(ANY_UID, ANY_GID),
592 its_remote_address, its_remote_port);
593 } else if (!utility::is_valid_message_type(tp::tp::tp_flag_unset(
594 _buffer[i + VSOMEIP_MESSAGE_TYPE_POS]))) {
595 VSOMEIP_ERROR << "use: Invalid message type: 0x"
596 << std::hex << std::setw(2) << std::setfill('0')
597 << std::uint32_t(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])
598 << " local: " << get_address_port_local()
599 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
600 } else if (!utility::is_valid_return_code(static_cast<return_code_e>(
601 _buffer[i + VSOMEIP_RETURN_CODE_POS]))) {
602 VSOMEIP_ERROR << "use: Invalid return code: 0x"
603 << std::hex << std::setw(2) << std::setfill('0')
604 << std::uint32_t(_buffer[i + VSOMEIP_RETURN_CODE_POS])
605 << " local: " << get_address_port_local()
606 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
607 } else if (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])
608 && get_local_port() == configuration_->get_sd_port()) {
609 VSOMEIP_WARNING << "use: Received a SomeIP/TP message on SD port:"
610 << " local: " << get_address_port_local()
611 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
612 }
613 return;
614 }
615 remaining_bytes -= current_message_size;
616 const service_t its_service = VSOMEIP_BYTES_TO_WORD(_buffer[i + VSOMEIP_SERVICE_POS_MIN],
617 _buffer[i + VSOMEIP_SERVICE_POS_MAX]);
618 if (utility::is_request(
619 _buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) {
620 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
621 _buffer[i + VSOMEIP_CLIENT_POS_MIN],
622 _buffer[i + VSOMEIP_CLIENT_POS_MAX]);
623 if (its_client != MAGIC_COOKIE_CLIENT) {
624 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
625 _buffer[i + VSOMEIP_SESSION_POS_MIN],
626 _buffer[i + VSOMEIP_SESSION_POS_MAX]);
627 clients_mutex_.lock();
628 clients_[its_client][its_session] = _remote;
629 clients_mutex_.unlock();
630 }
631 } else if (its_service != VSOMEIP_SD_SERVICE
632 && utility::is_notification(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])
633 && joined_group_) {
634 boost::system::error_code ec;
635 const auto found_address = joined_.find(_destination.to_string(ec));
636 if (found_address != joined_.end()) {
637 found_address->second = true;
638 }
639 }
640 if (tp::tp::tp_flag_is_set(_buffer[i + VSOMEIP_MESSAGE_TYPE_POS])) {
641 const method_t its_method = VSOMEIP_BYTES_TO_WORD(_buffer[i + VSOMEIP_METHOD_POS_MIN],
642 _buffer[i + VSOMEIP_METHOD_POS_MAX]);
643 if (!tp_segmentation_enabled(its_service, its_method)) {
644 VSOMEIP_WARNING << "use: Received a SomeIP/TP message for service: 0x" << std::hex << its_service
645 << " method: 0x" << its_method << " which is not configured for TP:"
646 << " local: " << get_address_port_local()
647 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
648 return;
649 }
650 const auto res = tp_reassembler_->process_tp_message(
651 &_buffer[i], current_message_size,
652 its_remote_address, its_remote_port);
653 if (res.first) {
654 if (utility::is_request(res.second[VSOMEIP_MESSAGE_TYPE_POS])) {
655 const client_t its_client = VSOMEIP_BYTES_TO_WORD(
656 res.second[VSOMEIP_CLIENT_POS_MIN],
657 res.second[VSOMEIP_CLIENT_POS_MAX]);
658 if (its_client != MAGIC_COOKIE_CLIENT) {
659 const session_t its_session = VSOMEIP_BYTES_TO_WORD(
660 res.second[VSOMEIP_SESSION_POS_MIN],
661 res.second[VSOMEIP_SESSION_POS_MAX]);
662 std::lock_guard<std::mutex> its_client_lock(clients_mutex_);
663 clients_[its_client][its_session] = _remote;
664 }
665 } else if (its_service != VSOMEIP_SD_SERVICE
666 && utility::is_notification(res.second[VSOMEIP_MESSAGE_TYPE_POS])
667 && joined_group_) {
668 boost::system::error_code ec;
669 const auto found_address = joined_.find(_destination.to_string(ec));
670 if (found_address != joined_.end()) {
671 found_address->second = true;
672 }
673 }
674 its_host->on_message(&res.second[0],
675 static_cast<std::uint32_t>(res.second.size()),
676 this, _destination, VSOMEIP_ROUTING_CLIENT,
677 std::make_pair(ANY_UID, ANY_GID),
678 its_remote_address, its_remote_port);
679 }
680 } else {
681 if (its_service != VSOMEIP_SD_SERVICE ||
682 (current_message_size > VSOMEIP_SOMEIP_HEADER_SIZE &&
683 current_message_size >= remaining_bytes)) {
684 its_host->on_message(&_buffer[i],
685 current_message_size, this, _destination,
686 VSOMEIP_ROUTING_CLIENT,
687 std::make_pair(ANY_UID, ANY_GID),
688 its_remote_address, its_remote_port);
689 } else {
690 //ignore messages for service discovery with shorter SomeIP length
691 VSOMEIP_ERROR << "Received an unreliable vSomeIP SD message with too short length field"
692 << " local: " << get_address_port_local()
693 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
694 }
695 }
696 i += current_message_size;
697 } else {
698 VSOMEIP_ERROR << "Received an unreliable vSomeIP message with bad length field"
699 << " local: " << get_address_port_local()
700 << " remote: " << its_remote_address << ":" << std::dec << its_remote_port;
701 if (remaining_bytes > VSOMEIP_SERVICE_POS_MAX) {
702 service_t its_service = VSOMEIP_BYTES_TO_WORD(_buffer[VSOMEIP_SERVICE_POS_MIN],
703 _buffer[VSOMEIP_SERVICE_POS_MAX]);
704 if (its_service != VSOMEIP_SD_SERVICE) {
705 if (read_message_size == 0) {
706 VSOMEIP_ERROR << "Ignoring unreliable vSomeIP message with SomeIP message length 0!";
707 } else {
708 auto its_endpoint_host = endpoint_host_.lock();
709 if (its_endpoint_host) {
710 its_endpoint_host->on_error(&_buffer[i],
711 (uint32_t)remaining_bytes, this,
712 its_remote_address, its_remote_port);
713 }
714 }
715 }
716 }
717 remaining_bytes = 0;
718 }
719 } while (remaining_bytes > 0);
720 }
721 }
722}
