207bool configuration_impl::load(const std::string &_name) {
208 std::lock_guard<std::mutex> its_lock(mutex_);
209 if (is_loaded_)
210 return true;
211
212 // Predefine file / folder
213 std::string its_file(VSOMEIP_DEFAULT_CONFIGURATION_FILE); // configuration file
214 std::string its_folder(VSOMEIP_DEFAULT_CONFIGURATION_FOLDER); // configuration folder
215
216 // Override with local file / folder (if existing)
217 std::string its_local_file(VSOMEIP_LOCAL_CONFIGURATION_FILE);
218 if (utility::is_file(its_local_file)) {
219 its_file = its_local_file;
220 }
221
222 std::string its_local_folder(VSOMEIP_LOCAL_CONFIGURATION_FOLDER);
223 if (utility::is_folder(its_local_folder)) {
224 its_folder = its_local_folder;
225 }
226
227 // Override with path from environment (if existing)
228 const char *its_env = getenv(VSOMEIP_ENV_CONFIGURATION);
229 if (nullptr != its_env) {
230 if (utility::is_file(its_env)) {
231 its_file = its_env;
232 its_folder = "";
233 } else if (utility::is_folder(its_env)) {
234 its_folder = its_env;
235 its_file = "";
236 }
237 }
238
239 std::set<std::string> its_input;
240 if (its_file != "") {
241 its_input.insert(its_file);
242 }
243 if (its_folder != "") {
244 its_input.insert(its_folder);
245#ifndef _WIN32
246 // load security configuration files from UID_GID sub folder if existing
247 std::stringstream its_security_config_folder;
248 its_security_config_folder << its_folder << "/" << getuid() << "_" << getgid();
249 if (utility::is_folder(its_security_config_folder.str())) {
250 its_input.insert(its_security_config_folder.str());
251 }
252#endif
253 }
254
255 // Determine standard configuration file
256 its_env = getenv(VSOMEIP_ENV_MANDATORY_CONFIGURATION_FILES);
257 if (nullptr != its_env) {
258 std::string its_temp(its_env);
259 set_mandatory(its_temp);
260 } else {
261 set_mandatory(VSOMEIP_MANDATORY_CONFIGURATION_FILES);
262 }
263
264 // Start reading
265 std::set<std::string> its_failed;
266
267 std::chrono::steady_clock::time_point begin = std::chrono::steady_clock::now();
268 std::vector<configuration_element> its_mandatory_elements;
269 std::vector<configuration_element> its_optional_elements;
270
271 // Dummy initialization; maybe we'll find no logging configuration
272 logger::logger_impl::init(shared_from_this());
273
274 // Look for the standard configuration file
275 read_data(its_input, its_mandatory_elements, its_failed, true);
276 load_data(its_mandatory_elements, true, false);
277
278 // If the configuration is incomplete, this is the routing manager configuration or
279 // the routing is yet unknown, read the full set of configuration files
280 if (its_mandatory_elements.empty() ||
281 _name == get_routing_host() ||
282 "" == get_routing_host()) {
283 read_data(its_input, its_optional_elements, its_failed, false);
284 load_data(its_mandatory_elements, false, true);
285 load_data(its_optional_elements, true, true);
286 }
287
288 // Tell, if reading of configuration file(s) failed.
289 // (This may file if the logger configuration is incomplete/missing).
290 for (auto f : its_failed)
291 VSOMEIP_WARNING << "Reading of configuration file \""
292 << f << "\" failed. Configuration may be incomplete.";
293
294 // set global unicast address for all services with magic cookies enabled
295 set_magic_cookies_unicast_address();
296
297 std::chrono::steady_clock::time_point end = std::chrono::steady_clock::now();
298 VSOMEIP_INFO << "Parsed vsomeip configuration in "
299 << std::chrono::duration_cast<std::chrono::milliseconds>(end - begin).count()
300 << "ms";
301
302 for (auto i : its_input) {
303 if (utility::is_file(i))
304 VSOMEIP_INFO << "Using configuration file: \"" << i << "\".";
305
306 if (utility::is_folder(i))
307 VSOMEIP_INFO << "Using configuration folder: \"" << i << "\".";
308 }
309
310 is_loaded_ = true;
311
312 return is_loaded_;
313}

322bool configuration_impl::remote_offer_info_add(service_t _service,
323 instance_t _instance,
324 std::uint16_t _port,
325 bool _reliable,
326 bool _magic_cookies_enabled) {
327 bool ret = false;
328 if (!is_loaded_) {
329 VSOMEIP_ERROR << __func__ << " shall only be called after normal"
330 "configuration has been parsed";
331 } else {
332 std::shared_ptr<service> its_service(std::make_shared<service>());
333 its_service->service_ = _service;
334 its_service->instance_ = _instance;
335 its_service->reliable_ = its_service->unreliable_ = ILLEGAL_PORT;
336 _reliable ?
337 its_service->reliable_ = _port :
338 its_service->unreliable_ = _port;
339 its_service->unicast_address_ = default_unicast_;
340 its_service->multicast_address_ = "";
341 its_service->multicast_port_ = ILLEGAL_PORT;
342 its_service->protocol_ = "someip";
343
344 {
345 std::lock_guard<std::mutex> its_lock(services_mutex_);
346 bool updated(false);
347 auto found_service = services_.find(its_service->service_);
348 if (found_service != services_.end()) {
349 auto found_instance = found_service->second.find(its_service->instance_);
350 if (found_instance != found_service->second.end()) {
351 VSOMEIP_INFO << "Updating remote configuration for service ["
352 << std::hex << std::setw(4) << std::setfill('0')
353 << its_service->service_ << "." << its_service->instance_ << "]";
354 if (_reliable) {
355 found_instance->second->reliable_ = its_service->reliable_;
356 } else {
357 found_instance->second->unreliable_ = its_service->unreliable_;
358 }
359 updated = true;
360 }
361 }
362 if (!updated) {
363 services_[_service][_instance] = its_service;
364 VSOMEIP_INFO << "Added new remote configuration for service ["
365 << std::hex << std::setw(4) << std::setfill('0')
366 << its_service->service_ << "."
367 << std::hex << std::setw(4) << std::setfill('0')
368 << its_service->instance_ << "]";
369 }
370 if (_magic_cookies_enabled) {
371 magic_cookies_[its_service->unicast_address_].insert(its_service->reliable_);
372 }
373 }
374 ret = true;
375 }
376 return ret;
377}

417void configuration_impl::read_data(const std::set<std::string> &_input,
418 std::vector<configuration_element> &_elements, std::set<std::string> &_failed,
419 bool _mandatory_only) {
420 for (auto i : _input) {
421 if (utility::is_file(i)) {
422 if (is_mandatory(i) == _mandatory_only) {
423 boost::property_tree::ptree its_tree;
424 try {
425 boost::property_tree::json_parser::read_json(i, its_tree);
426 _elements.push_back({ i, its_tree });
427 }
428 catch (boost::property_tree::json_parser_error &e) {
429 #ifdef _WIN32
430 e; // silence MSVC warning C4101
431 #endif
432 _failed.insert(i);
433 }
434 }
435 } else if (utility::is_folder(i)) {
436 boost::filesystem::path its_path(i);
437 for (auto j = boost::filesystem::directory_iterator(its_path);
438 j != boost::filesystem::directory_iterator();
439 j++) {
440 auto its_file_path = j->path();
441 if (!boost::filesystem::is_directory(its_file_path)) {
442 const std::string& its_name = its_file_path.string();
443 if (is_mandatory(its_name) == _mandatory_only) {
444 boost::property_tree::ptree its_tree;
445 try {
446 boost::property_tree::json_parser::read_json(its_name, its_tree);
447 _elements.push_back({its_name, its_tree});
448 }
449 catch (...) {
450 _failed.insert(its_name);
451 }
452 }
453 }
454 }
455 }
456 }

460bool configuration_impl::load_data(const std::vector<configuration_element> &_elements,
461 bool _load_mandatory, bool _load_optional) {
462 // Load logging configuration data
463 std::set<std::string> its_warnings;
464
465 if (!is_logging_loaded_) {
466 for (const auto& e : _elements)
467 is_logging_loaded_
468 = load_logging(e, its_warnings) || is_logging_loaded_;
469
470 if (is_logging_loaded_) {
471 logger::logger_impl::init(shared_from_this());
472 for (auto w : its_warnings)
473 VSOMEIP_WARNING << w;
474 }
475 }
476
477 bool has_routing(false);
478 bool has_applications(false);
479 if (_load_mandatory) {
480 // Load mandatory configuration data
481 for (const auto& e : _elements) {
482 has_routing = load_routing(e) || has_routing;
483 has_applications = load_applications(e) || has_applications;
484 load_network(e);
485 load_diagnosis_address(e);
486 load_shutdown_timeout(e);
487 load_payload_sizes(e);
488 load_endpoint_queue_sizes(e);
489 load_tcp_restart_settings(e);
490 load_permissions(e);
491 load_security(e);
492 load_tracing(e);
493 load_udp_receive_buffer_size(e);
494 }
495 }
496
497 if (_load_optional) {
498 for (const auto& e : _elements) {
499 load_unicast_address(e);
500 load_netmask(e);
501 load_device(e);
502 load_service_discovery(e);
503 load_npdu_default_timings(e);
504 load_services(e);
505 load_internal_services(e);
506 load_clients(e);
507 load_watchdog(e);
508 load_selective_broadcasts_support(e);
509 load_e2e(e);
510 load_debounce(e);
511 load_acceptances(e);
512 load_secure_services(e);
513 }
514 }
515
516 return is_logging_loaded_ && has_routing && has_applications;
517}

519bool configuration_impl::load_logging(
520 const configuration_element &_element, std::set<std::string> &_warnings) {
521 try {
522 auto its_logging = _element.tree_.get_child("logging");
523 for (auto i = its_logging.begin(); i != its_logging.end(); ++i) {
524 std::string its_key(i->first);
525 if (its_key == "console") {
526 if (is_configured_[ET_LOGGING_CONSOLE]) {
527 _warnings.insert("Multiple definitions for logging.console."
528 " Ignoring definition from " + _element.name_);
529 } else {
530 std::string its_value(i->second.data());
531 has_console_log_ = (its_value == "true");
532 is_configured_[ET_LOGGING_CONSOLE] = true;
533 }
534 } else if (its_key == "file") {
535 if (is_configured_[ET_LOGGING_FILE]) {
536 _warnings.insert("Multiple definitions for logging.file."
537 " Ignoring definition from " + _element.name_);
538 } else {
539 for (auto j : i->second) {
540 std::string its_sub_key(j.first);
541 std::string its_sub_value(j.second.data());
542 if (its_sub_key == "enable") {
543 has_file_log_ = (its_sub_value == "true");
544 } else if (its_sub_key == "path") {
545 logfile_ = its_sub_value;
546 }
547 }
548 is_configured_[ET_LOGGING_FILE] = true;
549 }
550 } else if (its_key == "dlt") {
551 if (is_configured_[ET_LOGGING_DLT]) {
552 _warnings.insert("Multiple definitions for logging.dlt."
553 " Ignoring definition from " + _element.name_);
554 } else {
555 std::string its_value(i->second.data());
556 has_dlt_log_ = (its_value == "true");
557 is_configured_[ET_LOGGING_DLT] = true;
558 }
559 } else if (its_key == "level") {
560 if (is_configured_[ET_LOGGING_LEVEL]) {
561 _warnings.insert("Multiple definitions for logging.level."
562 " Ignoring definition from " + _element.name_);
563 } else {
564 std::string its_value(i->second.data());
565 loglevel_
566 = (its_value == "trace" ?
567 vsomeip_v3::logger::level_e::LL_VERBOSE :
568 (its_value == "debug" ?
569 vsomeip_v3::logger::level_e::LL_DEBUG :
570 (its_value == "info" ?
571 vsomeip_v3::logger::level_e::LL_INFO :
572 (its_value == "warning" ?
573 vsomeip_v3::logger::level_e::LL_WARNING :
574 (its_value == "error" ?
575 vsomeip_v3::logger::level_e::LL_ERROR :
576 (its_value == "fatal" ?
577 vsomeip_v3::logger::level_e::LL_FATAL :
578 vsomeip_v3::logger::level_e::LL_INFO))))));
579 is_configured_[ET_LOGGING_LEVEL] = true;
580 }
581 } else if (its_key == "version") {
582 std::stringstream its_converter;
583 for (auto j : i->second) {
584 std::string its_sub_key(j.first);
585 std::string its_sub_value(j.second.data());
586 if (its_sub_key == "enable") {
587 log_version_ = (its_sub_value == "true");
588 } else if (its_sub_key == "interval") {
589 its_converter << std::dec << its_sub_value;
590 its_converter >> log_version_interval_;
591 }

666void configuration_impl::load_application_data(
667 const boost::property_tree::ptree &_tree, const std::string &_file_name) {
668 std::string its_name("");
669 client_t its_id(VSOMEIP_CLIENT_UNSET);
670 std::size_t its_max_dispatchers(VSOMEIP_MAX_DISPATCHERS);
671 std::size_t its_max_dispatch_time(VSOMEIP_MAX_DISPATCH_TIME);
672 std::size_t its_io_thread_count(VSOMEIP_IO_THREAD_COUNT);
673 std::size_t its_request_debounce_time(VSOMEIP_REQUEST_DEBOUNCE_TIME);
674 std::map<plugin_type_e, std::set<std::string>> plugins;
675 int its_io_thread_nice_level(VSOMEIP_IO_THREAD_NICE_LEVEL);
676 std::string its_overlay;
677#ifdef VSOMEIP_HAS_SESSION_HANDLING_CONFIG
678 bool has_session_handling(true);
679#endif // VSOMEIP_HAS_SESSION_HANDLING_CONFIG
680 for (auto i = _tree.begin(); i != _tree.end(); ++i) {
681 std::string its_key(i->first);
682 std::string its_value(i->second.data());
683 std::stringstream its_converter;
684 if (its_key == "name") {
685 its_name = its_value;
686 } else if (its_key == "id") {
687 if (its_value.size() > 1 && its_value[0] == '0' && its_value[1] == 'x') {
688 its_converter << std::hex << its_value;
689 } else {
690 its_converter << std::dec << its_value;
691 }
692 its_converter >> its_id;
693 } else if (its_key == "max_dispatchers") {
694 its_converter << std::dec << its_value;
695 its_converter >> its_max_dispatchers;
696 } else if (its_key == "max_dispatch_time") {
697 its_converter << std::dec << its_value;
698 its_converter >> its_max_dispatch_time;
699 } else if (its_key == "threads") {
700 its_converter << std::dec << its_value;
701 its_converter >> its_io_thread_count;
702 if (its_io_thread_count == 0) {
703 VSOMEIP_WARNING << "Min. number of threads per application is 1";
704 its_io_thread_count = 1;
705 } else if (its_io_thread_count > 255) {
706 VSOMEIP_WARNING << "Max. number of threads per application is 255";
707 its_io_thread_count = 255;
708 }
709 } else if (its_key == "io_thread_nice") {
710 its_converter << std::dec << its_value;
711 its_converter >> its_io_thread_nice_level;
712 } else if (its_key == "request_debounce_time") {
713 its_converter << std::dec << its_value;
714 its_converter >> its_request_debounce_time;
715 if (its_request_debounce_time > 10000) {
716 VSOMEIP_WARNING << "Max. request debounce time is 10.000ms";
717 its_request_debounce_time = 10000;
718 }
719 } else if (its_key == "plugins") {
720 plugins = load_plugins(i->second, its_name);
721 } else if (its_key == "overlay") {
722 its_overlay = its_value;
723 }
724#ifdef VSOMEIP_HAS_SESSION_HANDLING_CONFIG
725 else if (its_key == "has_session_handling") {
726 has_session_handling = (its_value != "false");
727 }
728#endif // VSOMEIP_HAS_SESSION_HANDLING_CONFIG
729 }
730 if (its_name != "") {
731 if (applications_.find(its_name) == applications_.end()) {
732 if (its_id != VSOMEIP_CLIENT_UNSET) {
733 if (!is_configured_client_id(its_id)) {
734 client_identifiers_.insert(its_id);
735 } else {
736 VSOMEIP_ERROR << "Multiple applications are configured to use"
737 << " client identifier " << std::hex << its_id
738 << ". Ignoring the configuration for application "
739 << its_name;
740 its_id = VSOMEIP_CLIENT_UNSET;
741 }
742 }
743 applications_[its_name]
744 = std::make_tuple(its_id, its_max_dispatchers,
745 its_max_dispatch_time, its_io_thread_count,
746 its_request_debounce_time, plugins, its_io_thread_nice_level,
747 its_overlay
748#ifdef VSOMEIP_HAS_SESSION_HANDLING_CONFIG
749 , has_session_handling
750#endif // VSOMEIP_HAS_SESSION_HANDLING_CONFIG
751 );
752 } else {
753 VSOMEIP_WARNING << "Multiple configurations for application "
754 << its_name << ". Ignoring a configuration from "
755 << _file_name;
756 }
757 }

968 vsomeip_v3::trace::match_t its_match;
969 load_trace_filter_match(i->second, its_match);
970 if (i->first == "from") {
971 _filter->is_range_ = true;
972 _filter->matches_.insert(_filter->matches_.begin(), its_match);

975 _filter->matches_.push_back(its_match);
976 }

982 const boost::property_tree::ptree &_data,
983 vsomeip_v3::trace::match_t &_match) {
984 std::stringstream its_converter;
985
986 if (_data.size() == 0) {
987 const std::string& its_value(_data.data());
988 service_t its_service(ANY_SERVICE);
989 if (its_value.find("0x") == 0) {
990 its_converter << std::hex << its_value;
991 } else {
992 its_converter << std::dec << its_value;
993 }
994 its_converter >> its_service;
995
996 std::get<0>(_match) = its_service;
997 std::get<1>(_match) = ANY_INSTANCE;
998 std::get<2>(_match) = ANY_METHOD;
999 } else {
1000 std::get<0>(_match) = ANY_SERVICE;
1001 std::get<1>(_match) = ANY_INSTANCE;
1002 std::get<2>(_match) = ANY_METHOD;
1003
1004 for (auto i = _data.begin(); i != _data.end(); ++i) {
1005 std::string its_value;
1006
1007 its_converter.str("");
1008 its_converter.clear();
1009
1010 try {
1011 its_value = i->second.data();
1012 if (its_value == "any") its_value = "0xffff";
1013
1014 if (i->first == "service") {
1015 service_t its_service(ANY_SERVICE);
1016 if (its_value.find("0x") == 0) {
1017 its_converter << std::hex << its_value;
1018 } else {
1019 its_converter << std::dec << its_value;
1020 }
1021 its_converter >> its_service;

1024 instance_t its_instance(ANY_INSTANCE);
1025 if (its_value.find("0x") == 0) {
1026 its_converter << std::hex << its_value;
1027 } else {
1028 its_converter << std::dec << its_value;
1029 }
1030 its_converter >> its_instance;
1031 std::get<1>(_match) = its_instance;
1032 } else if (i->first == "method") {
1033 method_t its_method(ANY_METHOD);
1034 if (its_value.find("0x") == 0) {
1035 its_converter << std::hex << its_value;
1036 } else {
1037 its_converter << std::dec << its_value;
1038 }
1039 its_converter >> its_method;
1040 std::get<2>(_match) = its_method;
1041 }
1042 } catch (...) {

1603 for (auto i = _tree.begin(); i != _tree.end(); ++i) {
1604 event_t its_event_id(0);
1605 bool its_is_field(false);
1606 reliability_type_e its_reliability(reliability_type_e::RT_UNKNOWN);
1607
1608 for (auto j = i->second.begin(); j != i->second.end(); ++j) {
1609 std::string its_key(j->first);
1610 std::string its_value(j->second.data());
1611 if (its_key == "event") {
1612 std::stringstream its_converter;
1613 if (its_value.size() > 1 && its_value[0] == '0' && its_value[1] == 'x') {
1614 its_converter << std::hex << its_value;
1615 } else {
1616 its_converter << std::dec << its_value;
1617 }
1618 its_converter >> its_event_id;
1619 } else if (its_key == "is_field") {
1620 its_is_field = (its_value == "true");
1621 } else if (its_key == "is_reliable") {
1622 if (its_value == "true")
1623 its_reliability = reliability_type_e::RT_RELIABLE;
1624 else
1625 its_reliability = reliability_type_e::RT_UNRELIABLE;
1626 }
1627 }
1628
1629 if (its_event_id > 0) {
1630 std::shared_ptr<event> its_event;
1631
1632 auto found_event = _service->events_.find(its_event_id);
1633 if (found_event != _service->events_.end()) {
1634 if (found_event->second->is_placeholder_) {
1635 its_event = found_event->second;
1636 } else {
1637 VSOMEIP_ERROR << "Multiple configurations for event ["
1638 << std::hex << _service->service_ << "."
1639 << _service->instance_ << "."
1640 << its_event_id << "]";
1641 }
1642 } else {
1643 its_event = std::make_shared<event>(its_event_id);
1644 _service->events_[its_event_id] = its_event;
1645 }
1646
1647 if (its_event) {
1648 // If event reliability type was not configured,
1649 if (its_reliability == reliability_type_e::RT_UNKNOWN) {
1650 if (_service->unreliable_ != ILLEGAL_PORT) {
1651 its_reliability = reliability_type_e::RT_UNRELIABLE;
1652 } else if (_service->reliable_ != ILLEGAL_PORT) {
1653 its_reliability = reliability_type_e::RT_RELIABLE;
1654 }
1655 VSOMEIP_WARNING << "Reliability type for event ["
1656 << std::hex << _service->service_ << "."
1657 << _service->instance_ << "."
1658 << its_event_id << "] was not configured Using : "
1659 << ((its_reliability == reliability_type_e::RT_RELIABLE)
1660 ? "RT_RELIABLE" : "RT_UNRELIABLE");
1661 }
1662
1663 its_event->is_placeholder_ = false;
1664 its_event->reliability_ = its_reliability;
1665 its_event->is_field_ = its_is_field;
1666 }
1667 }
1668 }

2349 service_t _service, instance_t _instance,
2350 uint16_t _remote_port, bool _reliable,
2351 std::map<bool, std::set<uint16_t> > &_used_client_ports,
2352 uint16_t &_client_port) const {

2367 }

2380 if (!is_configured) {
2381 // Neither specific not generic configurarion available,
2382 // use dynamic port configuration!
2383 _client_port = 0;

2385 }

2388 VSOMEIP_ERROR << "Cannot find free client port for communication to service: "

2390 << _instance << " remote_port: "
2391 << _remote_port << " reliable: "
